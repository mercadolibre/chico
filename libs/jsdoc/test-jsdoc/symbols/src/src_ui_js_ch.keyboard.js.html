1 (function () {
  2 	/**
  3 	 * Map with references to key codes.
  4 	 * @private
  5 	 * @name ch.Keyboard#codeMap
  6 	 * @type object
  7 	 */
  8 	var codeMap = {
  9 		"13": "ENTER",
 10 		"27": "ESC",
 11 		"37": "LEFT_ARROW",
 12 		"38": "UP_ARROW",
 13 		"39": "RIGHT_ARROW",
 14 		"40": "DOWN_ARROW",
 15 		 "8": "BACKSPACE"
 16 	};
 17 
 18 	/**
 19 	 * Keyboard event controller utility to know wich keys are begin.
 20 	 * @name Keyboard
 21 	 * @class Keyboard
 22 	 * @memberOf ch
 23 	 * @param event
 24 	 */
 25 	function keyboard(event) {
 26 		// Check for event existency on the map
 27 		if(!hasOwn(codeMap, event.keyCode)) { return; }
 28 
 29 		// Trigger custom event with original event as second parameter
 30 		$document.trigger(EVENT.KEY[codeMap[event.keyCode]], event);
 31 	}
 32 
 33 	exports.keyboard = keyboard;
 34 }());
