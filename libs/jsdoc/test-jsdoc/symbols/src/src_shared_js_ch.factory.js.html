1 (function () {
  2 	/**
  3 	 * Creational patterns to create UI Components
  4 	 * @methodOf ch
  5 	 * @param {Object} klass
  6 	 */
  7 	function factory(klass) {
  8 		// tiene que crear $.widget y $('').widget();
  9 		var name = klass.prototype.name,
 10 			map = {
 11 				'string': 'message',
 12 				'number': 'num',
 13 				'function': 'fn'
 14 			};
 15 
 16 		exports[klass.name] = klass;
 17 
 18 		// $.widget(options);
 19 		$[name] = function (options) {
 20 			return new klass(options);
 21 		};
 22 
 23 		// $(el).widget(options);
 24 		$.fn[name] = function (options) {
 25 			var widgets = [],
 26 				widget,
 27 				message = arguments[1],
 28 				type = typeof options;
 29 
 30 			if (options !== undefined && type !== 'object') {
 31 				var parameter = options;
 32 				options = {};
 33 				options[map[type]] = parameter;
 34 
 35 				// Could come a messages as a second argument
 36 				if (typeof message === 'string') {
 37 					options.message = message;
 38 				}
 39 			}
 40 
 41 			// http://docs.jquery.com/Plugins/Authoring
 42 			$.each(this, function (i, el) {
 43 				var $el = $(el),
 44 					data = $el.data(name);
 45 
 46 				if (!data) {
 47 					widget = new klass($el, options);
 48 					$el.data(name, widget);
 49 
 50 				} else {
 51 					widget = data;
 52 				}
 53 
 54 				widgets.push(widget);
 55 			});
 56 
 57 			return ((widgets.length > 1) ? widgets : widgets[0]);
 58 		};
 59 	}
 60 
 61 	exports.factory = factory;
 62 }());
