1 (function () {
  2 	/**
  3 	 * Returns a data object with features supported by the device
  4 	 * @name ch.support
  5 	 * @namespace
  6 	 */
  7 	var support = {};
  8 
  9 	/**
 10 	 * Verify that CSS Transitions are supported (or any of its browser-specific implementations).
 11 	 * @name transition
 12 	 * @memberOf ch.support
 13 	 * @type {Boolean}
 14 	 * @see <a href="http://gist.github.com/373874" target="_blank">http://gist.github.com/373874</a>
 15 	 */

 17 
 18 	/**
 19 	 * Boolean property that indicates if CSS "Fixed" positioning are supported by the device.
 20 	 * @name fixed
 21 	 * @memberOf ch.support
 22 	 * @type {Boolean}
 23 	 * @see <a href="http://kangax.github.com/cft/#IS_POSITION_FIXED_SUPPORTED" target="_blank">http://kangax.github.com/cft/#IS_POSITION_FIXED_SUPPORTED</a>
 24 	 */
 25 	support.fixed = (function () {
 26 
 27 		// Flag to know if position is supported
 28 		var isSupported = false,
 29 		// Create an element to test
 30 			el = document.createElement("div");
 31 
 32 		// Set the position fixed
 33 		el.style.position = "fixed";
 34 		// Set a top
 35 		el.style.top = "10px";
 36 
 37 		// Add element to DOM
 38 		body.appendChild(el);
 39 
 40 		// Compare setted offset with "in DOM" offset of element
 41 		isSupported = (el.offsetTop === 10);
 42 
 43 		// Delete element from DOM
 44 		body.removeChild(el);
 45 
 46 		// Results
 47 		return isSupported;
 48 	}());
 49 
 50 	exports.support = support;
 51 }());
