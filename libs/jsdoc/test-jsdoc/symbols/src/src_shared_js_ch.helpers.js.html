 99 	 */
100 	VENDOR_PREFIX = (function () {
101 
102 		var regex = /^(Webkit|Khtml|Moz|ms|O)(?=[A-Z])/,
103 			styleDeclaration = document.getElementsByTagName('script')[0].style,
104 			prop;
105 
106 		for (prop in styleDeclaration) {
107 			if (regex.test(prop)) {
108 				return prop.match(regex)[0].toLowerCase();
109 			}
110 		}
111 
112 		// Nothing found so far? Webkit does not enumerate over the CSS properties of the style object.
113 		// However (prop in style) returns the correct value, so we'll have to test for
114 		// the precence of a specific property
115 		if ('WebkitOpacity' in styleDeclaration) { return 'webkit'; }
116 		if ('KhtmlOpacity' in styleDeclaration) { return 'khtml'; }
117 
118 		return '';
119 	}()),
