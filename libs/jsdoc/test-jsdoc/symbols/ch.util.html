<static> {Object} ch.util.extend(obj, destination)
Copy all of properties from an object to a destination object, and returns the destination object. 
var Gizmo = {"name": "foo"};
ch.extend({
    "sayName": function () { console.log(this.name); },
    "foobar": "Some string"
}, Gizmo);

// Returns Gizmo
console.dir(gizmo);

// Gizmo Object
// foobar: "Some string"
// name: "foo"
// sayName: function () {}
// Add new funcionality to CH
ch.extend({
    "foobar": "Some string"
});

// Returns ch
console.dir(ch);
