158 /** 
159  * Utility to clone objects
160  * @function
161  * @name clon
162  * @param {Object} o Object to clone
163  * @return {Object}
164  * @memberOf ch
165  */
166 ch.clon = function(o) {
167     
168     obj = {};
169     
170     for (x in o) {
171         obj[x] = o[x]; 
172     };
173     
174     return obj;

178 /** 
179  * Class to create UI Components
180  * @abstract
181  * @name Factory
182  * @class Factory
183  * @param {Configuration Object} o 
184  * @example
185  *   o {
186  *      component: "chat",
187  *      callback: function(){},
188  *      [script]: "http://..",
189  *      [style]: "http://..",
190  *      [callback]: function(){}    
191  *   }
192  * @return {Collection} A collection of object instances
193  * @memberOf ch
194  */    
195 
196 ch.factory = function(o) {
197     
198     if (!o) { 
199         alert("Factory fatal error: Need and object {component:\"\"} to configure a component."); 
200         return;
201     };
202 
203     if (!o.component) { 
204         alert("Factory fatal error: No component defined."); 
205         return;
206     };
207 
208     var x = o.component;
209 
210     var create = function(x) { 
211 
212         // Send configuration to a component trough options object
213         $.fn[x] = function( options ) {
214 
215             var results = [];			    
216             var that = this;       
217 
218  			// Could be more than one argument
219  			var _arguments = arguments;
220  			
221             that.each( function(i, e) {
222             	
223 				var conf = options || {};
224 
225                 var context = {};
226                     context.type = x;
227                     context.element = e;
228                     context.$element = $(e);
229                     context.uid = ch.utils.index += 1; // Global instantiation index
230                
231 				switch(typeof conf) {
232 					// If argument is a number, join with the conf
233 					case "number":
234 						var num = conf;
235 						conf = {};
236 						conf.value = num;
237 						
238 						// Could come a messages as a second argument
239 		                if (_arguments[1]) {
240 		                    conf.msg = _arguments[1];
241 		                };
242 					break;
243 					
244 					// This could be a message
245 					case "string":						
246 						var msg = conf;
247 						conf = {};
248 						conf.msg = msg;
249 					break;
250 					
251 					// This is a condition for custom validation
252 					case "function":
253 						var func = conf;
254 						conf = {};
255 						conf.lambda = func;
256 						
257 						// Could come a messages as a second argument
258 		                if (_arguments[1]) {      	
259 		                    conf.msg = _arguments[1];
260 		                };
261 					break;
262 				};
263                
264                 // Create a component from his constructor
265                 var created = ch[x].call( context, conf );
266 
267 				/* 
268 					MAPPING INSTANCES
269 				
270     				Internal interface for avoid mapping objects
271     				{
272     					exists:true,
273     					object: {}
274     				}
275 				*/
276 				
277 				created = ( created.hasOwnProperty("public") ) ? created["public"] : created;
278 				
279 			    if (created.type) {
280 			        var type = created.type;		    
281                     // If component don't exists in the instances map create an empty array
282                     if (!ch.instances[type]) { ch.instances[type] = []; }
283                          ch.instances[type].push( created );
284 			    }
285                 
286                 // Avoid mapping objects that already exists
287 				if (created.exists) {				
288 					// Return the inner object
289 					created = created.object;
290 				}			
291 
292 			    results.push( created );
293 			    
294             });
295             
296             // return the created components collection or single component   
297             return ( results.length > 1 ) ? results : results[0];
298         };
299 
300         // if a callback is defined 
301         if ( o.callback ) { o.callback(); }
302                         
303     } // end create function
304     
305     if ( ch[o.component] ) {
306         // script already here, just create it
307         create(o.component);
308         
309     } else {
310         // get resurces and call create later
311         ch.get({
312             "method":"component",
313             "component":o.component,
314             "script": ( o.script )? o.script : "src/js/"+o.component+".js",
315             "styles": ( o.style ) ? o.style : "src/css/"+x+".css",
316             "callback":create
317         });
318     }
319 }

322 /**
323  * Load components or content
324  * @abstract
325  * @name Get
326  * @class Get
327  * @param o {Object} object 
328  * @example
329  *   o {
330  *      method: "content"|"component",
331  *      component: "chat",
332  *      [script]: "http://..",
333  *      [style]: "http://..",
334  *      [callback]: function(){}
335  *   }
336  * @memberOf ch
337  */
338 ch.get = function(o) {
339      
340     switch(o.method) {
341 		
342 		case "content":
343 
344 	        var that = o.that;
345 	        var conf = that.conf;
346 			var context = ( that.controller ) ? that.controller["public"] : that["public"];
347 
348 			// Set ajax config
349 			// On IE (6-7) "that" reference losts when I call ch.get for second time
350 			// Why?? I don't know... but with a setTimeOut() work fine!
351 			setTimeout(function(){
352 			
353 				$.ajax({
354 					url: conf.ajaxUrl,
355 					type: conf.ajaxType || 'GET',
356 					data: conf.ajaxParams,
357 					cache: true,
358 					async: true,
359 					beforeSend: function(jqXHR){
360 						jqXHR.setRequestHeader("X-Requested-With", "XMLHttpRequest");
361 					},
362 					success: function(data, textStatus, xhr){					
363 						that.$content.html( data );
364 						if ( conf.onContentLoad ) conf.onContentLoad.call(context);
365 						if ( conf.position ) ch.positioner(conf.position);
366 					},
367 					error: function(xhr, textStatus, errorThrown){
368 						data = (conf.hasOwnProperty("onContentError")) ? conf.onContentError.call(context, xhr, textStatus, errorThrown) : "<p>Error on ajax call </p>";
369 						that.$content.html( data );
370 						if ( conf.position ) ch.positioner(conf.position);
371 					}
372 				});
373 				
374 			}, 0);
375 			
376 		break;
377 	        
378 		case "component":
379 	        
380 	        // ch.get: "Should I get a style?"
381 	        if ( o.style ) {
382 	    		var style = document.createElement('link');
383 	        		style.href = o.style;
384 	    	    	style.rel = 'stylesheet';
385 	            	style.type = 'text/css';
386 	        }
387 	        // ch.get: "Should I get a script?"        
388 	        if ( o.script ) {
389 	    	   	var script = document.createElement("script");
390 	    			script.src = o.script;
391 	        }
392 	        
393 	        var head = document.getElementsByTagName("head")[0] || document.documentElement;
394 	
395 			// Handle Script loading
396 			var done = false;
397 	
398 			// Attach handlers for all browsers
399 			script.onload = script.onreadystatechange = function() {
400 	    
401 	    	if ( !done && (!this.readyState || 
402 						this.readyState === "loaded" || this.readyState === "complete") ) {
403 
404 					done = true;
405 
406 		   			// if callback is defined call it
407 		   	        if ( o.callback ) { o.callback( o.component ); }
408 
409 			   		// Handle memory leak in IE
410 		   			script.onload = script.onreadystatechange = null;
411 
412 			   		if ( head && script.parentNode ) { head.removeChild( script ); }
413 				}
414 			};
415 	            
416 			// Use insertBefore instead of appendChild to circumvent an IE6 bug.
417 			// This arises when a base node is used.
418 			if ( o.script ) { head.insertBefore( script, head.firstChild ); }
419 			if ( o.style ) { head.appendChild( style ); }
420 	    
421 		break;        
422 	}
423 
424 }

427 /**
428  * Returns a data object with features supported by the device
429  * @abstract
430  * @name Support
431  * @class Support
432  * @return {Object}
433  * @memberOf ch 
434  */
435 ch.support = function() {
436 	
437 	/**
438 	 * Private reference to the <body> element
439 	 * @private
440 	 * @name thisBody
441 	 * @type {HTMLBodyElement}
442 	 * @memberOf ch.Support
443 	 */
444 	var thisBody = document.body || document.documentElement;
445 	
446 	/**
447      * Based on: http://gist.github.com/373874
448      * Verify that CSS3 transition is supported (or any of its browser-specific implementations)
449      *
450      * @private
451      * @return {Boolean}
452      * @memberOf ch.Support
453      */
454 	var transition = (function(){
455 		var thisStyle = thisBody.style;
456 		return thisStyle.WebkitTransition !== undefined || thisStyle.MozTransition !== undefined || thisStyle.OTransition !== undefined || thisStyle.transition !== undefined;
457 	})();
458 
459     /**
460      * Based on: http://kangax.github.com/cft/#IS_POSITION_FIXED_SUPPORTED
461      * Verify that position fixed is supported
462      * 
463      * @private
464      * @return {Boolean}
465      * @memberOf ch.Support
466      */	
467 	var fixed = (function(){
468         var isSupported = false;
469 		var e = document.createElement("div");
470 			e.style.position = "fixed";
471 			e.style.top = "10px";
472 			
473 		thisBody.appendChild(e);
474 		if (e.offsetTop === 10) { isSupported = true; };
475   		thisBody.removeChild(e);
476   		
477   		return isSupported;
478   		
479 	})();
480 
481 	return {
482         /**
483          * Boolean property that indicates if CSS3 Transitions are supported by the device.
484          * @public
485          * @name transition
486          * @type {Boolean}
487          * @memberOf ch.Support
488          */
489  		transition: transition,
490         /**
491          * Boolean property that indicates if Fixed positioning are supported by the device.
492          * @public
493          * @name fixed
494          * @type {Boolean}
495          * @memberOf ch.Support
496          */
497 		fixed: fixed
498 	};
499
