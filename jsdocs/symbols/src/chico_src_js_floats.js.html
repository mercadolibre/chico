  2 /**
  3  * Abstract class of all floats UI-Objects.
  4  * @abstract
  5  * @name Floats
  6  * @class Floats
  7  * @augments ch.Object
  8  * @memberOf ch
  9  * @param {Configuration Object} conf Object with configuration properties
 10  * @returns {Chico-UI Object}
 11  * @see ch.Tooltip
 12  * @see ch.Layer
 13  * @see ch.Modal
 14  */ 
 15 
 16 ch.floats = function() {
 17 
 18     /**
 19      * Reference to a internal component instance, saves all the information and configuration properties.
 20      * @private
 21      * @name that
 22      * @type {Object}
 23      * @memberOf ch.Floats
 24      */ 
 25  	var that = this;
 26 	var conf = that.conf;
 27 
 28 /**
 29  *  Inheritance
 30  */
 31 
 32     that = ch.object.call(that);
 33     that.parent = ch.clon(that);
 34     
 35 /**
 36  *  Private Members
 37  */
 38 	var createCone = function() {
 39 		$("<div>")
 40 			.addClass("ch-cone")
 41 			.prependTo( that.$container );
 42 		
 43 		return;
 44 	};
 45 
 46 	var createClose = function() {
 47 		// Close Button
 48 		$("<div>")
 49 			.addClass("btn close")
 50 			.css("z-index", ch.utils.zIndex ++)
 51 			.bind("click", function(event){ that.hide(event); })
 52 			.prependTo( that.$container );
 53 		
 54 		// ESC key
 55 		ch.utils.document.bind(ch.events.KEY.ESC, function(event){ that.hide(event); });
 56 		
 57 		return;
 58 	};
 59 
 60     var createLayout = function() {
 61 		
 62 		that.$content = $("<div>")
 63 		    	.addClass("ch-" + that.type + "-content")
 64 		    	.html( that.loadContent() );
 65 		
 66 		that.$container = $("<div>")
 67 			.addClass("ch-" + that.type)
 68 			.css("z-index", ch.utils.zIndex ++)
 69 			.append( that.$content )
 70 			.appendTo("body");
 71 		
 72 		// Visual configuration
 73 		if( ch.utils.hasOwn(conf, "classes") ) that.$container.addClass(conf.classes);
 74 		if( ch.utils.hasOwn(conf, "width") ) that.$container.css("width", conf.width);
 75 		if( ch.utils.hasOwn(conf, "height") ) that.$container.css("height", conf.height);
 76 		if( ch.utils.hasOwn(conf, "closeButton") && conf.closeButton ) createClose();
 77 		if( ch.utils.hasOwn(conf, "cone") ) createCone();
 78 		if( ch.utils.hasOwn(conf, "fx") ) conf.fx = conf.fx; else conf.fx = true;
 79 		
 80 		// Cache - Default: true
 81 		conf.cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
 82 		
 83 		// Show component with effects
 84 		if( conf.fx ) {
 85 			that.$container.fadeIn("fast", function(){ that.callbacks("onShow"); });
 86 		
 87 		// Show component without effects
 88 		} else {
 89 			// TODO: that.$container.removeClass("ch-hide");
 90 			that.$container.show();
 91 			that.callbacks("onShow");
 92 		};
 93 		
 94 		// Position component
 95 		conf.position = conf.position || {};
 96 		conf.position.element = that.$container;
 97 		conf.position.hold = conf.hold || false;
 98 		ch.positioner.call(that);
 99 		
100 		return;
101     };
102     
103 
104 /**
105  *  Protected Members
106  */ 
107 			
108 /**
109  *  Public Members
110  */
111  
112 	that.active = false;
113 
114     /**
115      * Shows component's content.
116      * @public
117      * @name show
118      * @returns {Chico-UI Object}
119      * @memberOf ch.Floats
120      */ 
121 	that.show = function(event) {
122 		
123 		if ( event ) that.prevent(event);
124 		
125 		if ( that.active ) return;
126 		
127 		that.active = true;
128 		
129 		// Show if exist, else create		
130 		if ( that.$container ) {
131 				
132 			// If not cache... get content again! // Flush cache where?? when?? do it!
133 			if ( !conf.cache ) that.$content.html( that.loadContent() );
134 			
135 			// Detach the content of BODY
136 			var content = conf.content || conf.msg;
137 			if ( ch.utils.isSelector(content) ) $(content).detach();
138 
139     		that.$container
140     		    .appendTo("body")
141     			.css("z-index", ch.utils.zIndex++)
142 			    .fadeIn('fast', function(){ 
143 					that.active = true;
144 					
145 					// Callback execute
146 					that.callbacks('onShow');
147 				});
148 
149 			that.position("refresh");
150 						
151 			return that;
152 		};
153 		
154 		// If you reach here, create a float
155         createLayout();
156         
157         return that;
158 	};
159 
160     /**
161      * Hides component's content.
162      * @public
163      * @name hide
164      * @returns {Chico-UI Object}
165      * @memberOf ch.Floats
166      */ 
167 	that.hide = function(event) {
168 
169 		if (event) that.prevent(event);
170 		
171 		if (!that.active) return;
172 
173 		var afterHide = function(){ 
174 			 
175 			that.active = false;
176 			
177 			// Append the content of BODY
178 			var content = conf.content || conf.msg;
179 			
180 			if (ch.utils.isSelector(content)) {
181 
182 				if ($("body " + content + ".ch-hide").length > 0) return false;
183 
184 				that.$content.children()
185 					.clone()
186 					.addClass("ch-hide")
187 					.appendTo("body");
188 			};
189 			
190 			// Callback execute
191 			that.callbacks('onHide');
192 			
193 			that.$container.detach();
194 			
195 		};
196 		
197 		// Show component with effects
198 		if( conf.fx ) {
199 			that.$container.fadeOut("fast", afterHide);
200 		
201 		// Show component without effects
202 		} else {
203 			// TODO: that.$container.addClass("ch-hide");
204 			that.$container.hide();
205 			afterHide();
206 		};
207 		
208 		return that;
209 
210 	};
211 	
212 	return that;
213 	
214 };
