  2 /**
  3  * Create custom validation interfaces for Watcher validation engine.
  4  * @name Custom
  5  * @class Custom
  6  * @augments ch.Watcher
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  * @example
 11  * // Validate a even number
 12  * $("input").custom(function(value){
 13  *      return (value%2==0) ? true : false;
 14  * }, "Enter a even number");
 15  * @see ch.Watcher
 16  */
 17 
 18 ch.custom = function(conf) {
 19 
 20 /**
 21  *  Validation
 22  */	
 23 	
 24 	if (!conf.lambda) {
 25         alert("Custom Validation fatal error: Need a function to evaluate, try $().custom(function(){},\"Message\");");
 26     };
 27 
 28 /**
 29  *  Constructor
 30  */
 31 
 32 	conf = conf || {};
 33     conf.messages = conf.messages || {};
 34 
 35     if ( conf.hasOwnProperty("msg") ) { 
 36     	conf.messages.custom = conf.msg;
 37     	conf.msg = null;
 38     	delete conf.msg;
 39     };
 40     
 41 	// Define the validation interface    
 42     conf.custom = true;
 43 
 44 	// Add validation types
 45 	conf.types = "custom";
 46     // Define the conditions of this interface
 47     conf.conditions = [{
 48 		// I don't have pre-conditions, comes within conf.lambda argument 
 49         name: "custom",
 50         func: conf.lambda 
 51     }];
 52 
 53 
 54 	return ch.watcher.call(this, conf);
 55     
 56 };
