  2 /**
  3  * TabNavigator UI-Component for static and dinamic content.
  4  * @name TabNavigator
  5  * @class TabNavigator
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11 
 12 ch.tabNavigator = function(conf){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.TabNavigator
 20      */
 21 	var that = this;
 22 
 23 	that.$element.addClass('ch-tabNavigator');
 24 		
 25 	conf = ch.clon(conf);
 26 	conf.selected = conf.selected || conf.value || 0;
 27 	
 28 	that.conf = conf;
 29 	
 30 /**
 31  *	Inheritance
 32  */
 33 
 34     that = ch.controllers.call(that);
 35     that.parent = ch.clon(that);
 36 
 37 /**
 38  *  Private Members
 39  */
 40 
 41     /**
 42      * Reference to the tabNavigator's triggers.
 43      * @private
 44      * @name ul
 45      * @type {jQuery Object}
 46      * @memberOf ch.TabNavigator
 47      */
 48 	var ul = that.$element.children(':first').addClass('ch-tabNavigator-triggers');
 49     /**
 50      * The actual location hash, is used to know if there's a specific tab selected.
 51      * @private
 52      * @name hash
 53      * @type {String}
 54      * @memberOf ch.TabNavigator
 55      */
 56 	var hash = window.location.hash.replace("#!", "");
 57     /**
 58      * A boolean property to know if the some tag should be selected.
 59      * @private
 60      * @name hashed
 61      * @type {Boolean}
 62      * @default false
 63      * @memberOf ch.TabNavigator
 64      */
 65     var hashed = false;
 66     /**
 67      * Get wich tab is selected.
 68      * @private
 69      * @name selected
 70      * @type {Number}
 71      * @memberOf ch.TabNavigator
 72      */
 73     var selected = conf.selected;
 74 
 75 /**
 76  *  Protected Members
 77  */ 
 78  
 79  	that.$trigger = ul.find('a');
 80 	that.$content = ul.next().addClass('ch-tabNavigator-content box');
 81 	
 82 	that.select = function(tab){		
 83 		
 84 		selected = parseInt(tab);
 85 		
 86 		tab = that.children[selected];
 87 		
 88 		if(tab.active) return; // Don't click me if I'm open
 89 
 90 		// Hide my open bro
 91 		$.each(that.children, function(i, e){
 92 			if( e.active ) e.hide();
 93 		});
 94         
 95         tab.shoot();
 96         
 97         //Change location hash
 98 		window.location.hash = "#!" + tab.$content.attr("id");		
 99 		
100 		// Callback
101 		that.callbacks("onSelect");
102 		
103         return that;
104 	};
105     
106 /**
107  *  Public Members
108  */
109 
110     /**
111      * The component's instance unique identifier.
112      * @public
113      * @name uid
114      * @type {Number}
115      * @memberOf ch.TabNavigator
116      */
117 	that["public"].uid = that.uid;
118     /**
119      * The element reference.
120      * @public
121      * @name element
122      * @type {HTMLElement}
123      * @memberOf ch.TabNavigator
124      */
125 	that["public"].element = that.element;
126     /**
127      * The component's type.
128      * @public
129      * @name type
130      * @type {String}
131      * @memberOf ch.TabNavigator
132      */
133 	that["public"].type = that.type;
134     /**
135      * Children instances associated to this controller.
136      * @public
137      * @name children
138      * @type {Collection}
139      * @memberOf ch.TabNavigator
140      */
141 	that["public"].children = that.children;
142     /**
143      * Select a specific child.
144      * @public
145      * @function
146      * @name select
147      * @param {Number} tab Tab's index.
148      * @memberOf ch.TabNavigator
149      */
150 	that["public"].select = function(tab){
151 		that.select(tab);
152 		
153 		return that["public"];
154 	};
155     /**
156      * Returns the selected child's index.
157      * @public
158      * @function
159      * @name getSelected
160      * @return {Number} selected Tab's index.
161      * @memberOf ch.TabNavigator
162      */	
163 	that["public"].getSelected = function(){ return selected; };
164 	
165 /**
166  *  Default event delegation
167  */	
168     
169 	// Create children
170 	$.each(that.$trigger, function(i, e){
171 		var tab = {};
172 			tab.uid = that.uid + "#" + i;
173 			tab.type = "tab";
174 			tab.element = e;
175 			tab.$element = $(e);
176 			
177 		that.children.push( ch.tab.call(tab, that) );
178 	});
179 	
180 	//Default: Load hash tab or Open first tab	
181 	for(var i = that.children.length; i--; ){
182 		if ( that.children[i].$content.attr("id") === hash ) {
183 			that.select(i);
184 			
185 			hashed = true;
186 			
187 			break;
188 		};
189 	};
190 
191 	if ( !hashed ) that.children[conf.selected].shoot();
192 
193 	return that;
194 	
195 };

199 /**
200  * Simple unit of content for TabNavigators.
201  * @abstract
202  * @name Tab
203  * @class Tab
204  * @augments ch.Navs
205  * @memberOf ch
206  * @param {Configuration Object} conf Object with configuration properties
207  * @return {Chico-UI Object}
208  */
209 
210 ch.tab = function(controller){
211 
212     /**
213      * Reference to a internal component instance, saves all the information and configuration properties.
214      * @private
215      * @name that
216      * @type {Object}
217      * @memberOf ch.Tab
218      */
219  	var that = this;
220 	
221 	conf = {};
222 	if ( controller.conf.hasOwnProperty("onContentLoad") ) conf.onContentLoad = controller.conf.onContentLoad;
223 	if ( controller.conf.hasOwnProperty("onContentError") ) conf.onContentError = controller.conf.onContentError;	
224 	
225 	that.conf = conf;
226 /**
227  *	Inheritance
228  */
229 
230     that = ch.navs.call(that);
231     that.parent = ch.clon(that);
232 	that.controller = controller;
233 
234 /**
235  *  Private Members
236  */
237 	
238 	
239 /**
240  *  Protected Members
241  */ 
242     /**
243      * Reference to the trigger element.
244      * @private
245      * @name $trigger
246      * @type {jQuery Object}
247      * @memberOf ch.Tab
248      */
249 	that.$trigger = that.$element.addClass("ch-tabNavigator-trigger");
250 
251     /**
252      * The component's content.
253      * @private
254      * @name $content
255      * @type {jQuery Object}
256      * @memberOf ch.Tab
257      */	
258 	that.$content = (function(){
259 		
260 		var content = controller.$element.find("#" + that.element.href.split("#")[1]);
261 		
262 		// If there are a tabContent...
263 		if ( content.length > 0 ) {
264 			
265 			return content;
266 		
267 		// If tabContent doesn't exists        
268 		} else {
269 			// Set ajax configuration
270 			conf.ajax = true;
271 
272 			// Create tabContent
273 			return $("<div>")
274 				.attr("id", (that.element.href.split("#").length == 2) ? that.element.href.split("#")[1] : "ch-tab" + that.uid.replace("#","-") )
275 				.addClass("ch-hide")
276 				.appendTo( controller.$content );
277 		}; 
278 
279 	})();
280 
281     /**
282      * Process the show event.
283      * @private
284      * @function
285      * @name shoot
286      * @return {jQuery Object}
287      * @memberOf ch.Tab
288      */ 
289 	that.shoot = function(event){
290 		that.prevent(event);
291 
292 		// Load my content if I'need an ajax request 
293 		if( that.$content.html() == "" ) that.$content.html( that.loadContent() );
294 
295 		// Show me
296 		that.show(event);
297 		
298 		return that;
299 	};
300 
301 /**
302  *  Public Members
303  */
304 	
305 	
306 /**
307  *  Default event delegation
308  */	 	
309 	
310 	// Hide my content if im inactive
311 	if(!that.active) that.$content.addClass("ch-hide");
312 
313 	that.$trigger.bind('click', function(event){
314 		that.prevent(event);
315 		controller.select(that.uid.split("#")[1]);
316 	});
317 	
318 	return that;
319 }
