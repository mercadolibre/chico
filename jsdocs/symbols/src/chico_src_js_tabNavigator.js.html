  2 /**
  3  * TabNavigator UI-Component for static and dinamic content.
  4  * @name TabNavigator
  5  * @class TabNavigator
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11 
 12 ch.tabNavigator = function(conf){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.TabNavigator
 20      */
 21 	var that = this;
 22 
 23 	that.$element.addClass('ch-tabNavigator');
 24 		
 25 	conf = ch.clon(conf);
 26 	conf.selected = conf.selected || conf.value || 0;
 27 	
 28 	that.conf = conf;
 29 	
 30 /**
 31  *	Inheritance
 32  */
 33 
 34     that = ch.controllers.call(that);
 35     that.parent = ch.clon(that);
 36 
 37 /**
 38  *  Private Members
 39  */
 40 
 41     /**
 42      * Reference to the tabNavigator's triggers.
 43      * @private
 44      * @name ul
 45      * @type {jQuery Object}
 46      * @memberOf ch.TabNavigator
 47      */
 48 	var ul = that.$element.children(':first').addClass('ch-tabNavigator-triggers');
 49     /**
 50      * The actual location hash, is used to know if there's a specific tab selected.
 51      * @private
 52      * @name hash
 53      * @type {String}
 54      * @memberOf ch.TabNavigator
 55      */
 56 	var hash = window.location.hash.replace("#!", "");
 57     /**
 58      * A boolean property to know if the some tag should be selected.
 59      * @private
 60      * @name hashed
 61      * @type {Boolean}
 62      * @default false
 63      * @memberOf ch.TabNavigator
 64      */
 65     var hashed = false;
 66     /**
 67      * Get wich tab is selected.
 68      * @private
 69      * @name selected
 70      * @type {Number}
 71      * @memberOf ch.TabNavigator
 72      */
 73     var selected = conf.selected;
 74 
 75 /**
 76  *  Protected Members
 77  */ 
 78  
 79  	that.$trigger = ul.find('a');
 80 	that.$content = ul.next().addClass('ch-tabNavigator-content box');
 81 	
 82 	that.select = function(tab){		
 83 		
 84 		selected = parseInt(tab);
 85 		
 86 		tab = that.children[selected];
 87 		
 88 		if(tab.active) return; // Don't click me if I'm open
 89 
 90 		// Hide my open bro
 91 		$.each(that.children, function(i, e){
 92 			if( e.active ) e.hide();
 93 		});
 94         
 95         tab.shoot();
 96         
 97         //Change location hash
 98 		window.location.hash = "#!" + tab.$content.attr("id");		
 99 		
100 		// Callback
101 		that.callbacks("onSelect");
102 		
103         return that;
104 	};
105     
106 /**
107  *  Public Members
108  */
109 
110     /**
111      * The component's instance unique identifier.
112      * @public
113      * @name uid
114      * @type {Number}
115      * @memberOf ch.TabNavigator
116      */
117 	that["public"].uid = that.uid;
118     /**
119      * The element reference.
120      * @public
121      * @name element
122      * @type {HTMLElement}
123      * @memberOf ch.TabNavigator
124      */
125 	that["public"].element = that.element;
126     /**
127      * The component's type.
128      * @public
129      * @name type
130      * @type {String}
131      * @memberOf ch.TabNavigator
132      */
133 	that["public"].type = that.type;
134     /**
135      * Children instances associated to this controller.
136      * @public
137      * @name children
138      * @type {Collection}
139      * @memberOf ch.TabNavigator
140      */
141 	that["public"].children = that.children;
142     /**
143      * Select a specific child.
144      * @public
145      * @function
146      * @name select
147      * @param {Number} tab Tab's index.
148      * @memberOf ch.TabNavigator
149      */
150 	that["public"].select = function(tab){
151 		that.select(tab);
152 		
153 		return that["public"];
154 	};
155     /**
156      * Returns the selected child's index.
157      * @public
158      * @function
159      * @name getSelected
160      * @return {Number} selected Tab's index.
161      * @memberOf ch.TabNavigator
162      */	
163 	that["public"].getSelected = function(){ return selected; };
164 	
165 /**
166  *  Default event delegation
167  */	
168     
169 	// Create children
170 	$.each(that.$trigger, function(i, e){
171 		var tab = {};
172 			tab.uid = that.uid + "#" + i;
173 			tab.type = "tab";
174 			tab.element = e;
175 			tab.$element = $(e);
176 			
177 		that.children.push( ch.tab.call(tab, that) );
178 	});
179 	
180 	//Default: Load hash tab or Open first tab	
181 	for(var i = that.children.length; i--; ){
182 		if ( that.children[i].$content.attr("id") === hash ) {
183 			that.select(i);
184 			
185 			hashed = true;
186 			
187 			break;
188 		};
189 	};
190 
191 	if ( !hashed ) that.children[conf.selected].shoot();
192 
193 	return that;
194 	
195 };

199 /**
200  * Simple unit of content for TabNavigators.
201  * @name Tab
202  * @class Tab
203  * @augments ch.Navs
204  * @memberOf ch
205  * @param {Configuration Object} conf Object with configuration properties
206  * @return {Chico-UI Object}
207  */
208 
209 ch.tab = function(controller){
210 
211     /**
212      * Reference to a internal component instance, saves all the information and configuration properties.
213      * @private
214      * @name that
215      * @type {Object}
216      * @memberOf ch.Tab
217      */
218  	var that = this;
219 	
220 	conf = {};
221 	if ( controller.conf.hasOwnProperty("onContentLoad") ) conf.onContentLoad = controller.conf.onContentLoad;
222 	if ( controller.conf.hasOwnProperty("onContentError") ) conf.onContentError = controller.conf.onContentError;	
223 	
224 	that.conf = conf;
225 /**
226  *	Inheritance
227  */
228 
229     that = ch.navs.call(that);
230     that.parent = ch.clon(that);
231 	that.controller = controller;
232 
233 /**
234  *  Private Members
235  */
236 	
237 	
238 /**
239  *  Protected Members
240  */ 
241     /**
242      * Reference to the trigger element.
243      * @private
244      * @name $trigger
245      * @type {jQuery Object}
246      * @memberOf ch.Tab
247      */
248 	that.$trigger = that.$element.addClass("ch-tabNavigator-trigger");
249 
250     /**
251      * The component's content.
252      * @private
253      * @name $content
254      * @type {jQuery Object}
255      * @memberOf ch.Tab
256      */	
257 	that.$content = (function(){
258 		
259 		var content = controller.$element.find("#" + that.element.href.split("#")[1]);
260 		
261 		// If there are a tabContent...
262 		if ( content.length > 0 ) {
263 			
264 			return content;
265 		
266 		// If tabContent doesn't exists        
267 		} else {
268 			// Set ajax configuration
269 			conf.ajax = true;
270 
271 			// Create tabContent
272 			return $("<div>")
273 				.attr("id", (that.element.href.split("#").length == 2) ? that.element.href.split("#")[1] : "ch-tab" + that.uid.replace("#","-") )
274 				.addClass("ch-hide")
275 				.appendTo( controller.$content );
276 		}; 
277 
278 	})();
279 
280     /**
281      * Process the show event.
282      * @private
283      * @function
284      * @name shoot
285      * @return {jQuery Object}
286      * @memberOf ch.Tab
287      */ 
288 	that.shoot = function(event){
289 		that.prevent(event);
290 
291 		// Load my content if I'need an ajax request 
292 		if( that.$content.html() == "" ) that.$content.html( that.loadContent() );
293 
294 		// Show me
295 		that.show(event);
296 		
297 		return that;
298 	};
299 
300 /**
301  *  Public Members
302  */
303 	
304 	
305 /**
306  *  Default event delegation
307  */	 	
308 	
309 	// Hide my content if im inactive
310 	if(!that.active) that.$content.addClass("ch-hide");
311 
312 	that.$trigger.bind('click', function(event){
313 		that.prevent(event);
314 		controller.select(that.uid.split("#")[1]);
315 	});
316 	
317 	return that;
318 }
