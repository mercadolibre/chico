  2 /**
  3  * Positioner is a utility that resolve positioning problem for all UI-Objects.
  4  * @abstract
  5  * @name Positioner
  6  * @class Positioner
  7  * @memberOf ch
  8  * @param {Position Object} o Object with positioning properties
  9  * @return {jQuery Object}
 10  * @example
 11  * // First example
 12  * ch.positioner({
 13  *     element: $("#element1"),
 14  *     context: $("#context1"),
 15  *     points: "lt rt"                //  Element left-top point = Context left-bottom point
 16  * });
 17  * @example  
 18  * // Second example
 19  * ch.positioner({
 20  *     element: $("#element2"),
 21  *     context: $("#context2"),
 22  *     points: "lt lb"                //  Element center-middle point = Context center-middle point
 23  * });
 24  */
 25  
 26 ch.positioner = function(o) {
 27 
 28     /**
 29      * Constructs a new positioning, get viewport size, check for relative parent's offests, 
 30      * find the context and set the position to a given element.
 31      * @constructs
 32      * @private
 33      * @function
 34      * @name initPosition
 35      * @memberOf ch.Positioner
 36      */
 37     var initPosition = function(){
 38         viewport = getViewport();
 39         parentRelative = getParentRelative();
 40         context = getContext();
 41         setPosition();        
 42     };

 45     /**
 46      * Object that contains all properties for positioning
 47      * @private
 48      * @name o
 49      * @type {Position Object}
 50      * @example
 51      * ch.Positioner({
 52      *   element: $element
 53      *   [context]: $element | viewport
 54      *   [points]: "cm cm"
 55      *   [offset]: "x y" 
 56      *   [hold]: false
 57      * });
 58      * @memberOf ch.Positioner
 59      */
 60 	var o = o || this.conf.position;
 61         o.points = o.points || "cm cm";
 62         o.offset = o.offset || "0 0";

 64     /**
 65      * Reference to the DOM Element beign positioned
 66      * @private
 67      * @name element
 68      * @type {jQuery Object}
 69      * @memberOf ch.Positioner
 70      */
 71 	var element = $(o.element);
 72 		element.css("position","absolute");

 75      * Reference to the DOM Element that we will use as a reference
 76      * @private
 77      * @name context
 78      * @typeÂ {HTMLElement}
 79      * @memberOf ch.Positioner
 80      */
 81 	var context;

 95      * @name parentRelative

 98 	var parentRelative;

100     /**
101      * A map to reference the input points to output className
102      * @private
103      * @name classReferences
104      * @memberOf ch.Positioner
105      */
106     var classReferences = {
107 		"lt lb": "bottom",
108 		"lb lt": "top",
109 		"rt rb": "bottom",
110 		"rb rt": "top",
111 		"lt rt": "right",
112 		"cm cm": "center"
113 	};

115     /**
116      * Array with offset information
117      * @private
118      * @name splittedOffset
119      * @memberOf ch.Positioner
120      */
121     var splittedOffset = o.offset.split(" ");
122    	/**
123      * String with left offset information
124      * @private
125      */
126    	var offset_left = parseInt(splittedOffset[0]);
127    	/**
128      * String with top offset information
129      * @private
130      */
131     var offset_top = parseInt(splittedOffset[1]);
132 
133     /**
134      * Get the viewport size
135      * @private
136      * @function
137      * @name getViewport
138      * @return {Viewport Object}
139      * @memberOf ch.Positioner
140      */
141 	var getViewport = function() {
142         
143         var viewport, width, height, left, top, pageX, pageY, scrollBar = 30;	    
144 	    	    
145         // the more standards compliant browsers (mozilla/netscape/opera/IE7) use window.innerWidth and window.innerHeight
146 	    if (typeof window.innerWidth != "undefined") {
147 		    viewport = window;
148 			width = viewport.innerWidth - scrollBar;
149 			height = viewport.innerHeight;
150 			pageX = viewport.pageXOffset;
151 			pageY = viewport.pageYOffset;
152 
153 			// Return viewport object
154 			return {
155 				element: viewport,			
156 				left: 0 + offset_left + pageX - scrollBar,
157 				top: 0 + offset_top + pageY,
158 				bottom: height + pageY,
159 				right: width + pageX,
160 				width: width,
161 				height: height
162 			}
163 		}
164         // IE6 in standards compliant mode (i.e. with a valid doctype as the first line in the document)
165         // older versions of IE - viewport = document.getElementsByTagName('body')[0];		
166 		else {
167 			viewport = document.documentElement;
168 			width = viewport.clientWidth - scrollBar;
169 			height = viewport.clientHeight;
170 			pageX = viewport.scrollLeft;
171 			pageY = viewport.scrollTop;
172 			
173 			// Return viewport object
174 			return {
175 				element: viewport,			
176 				left: 0 + offset_left + pageX,
177 				top: 0 + offset_top + pageY,
178 				bottom: height + pageY,
179 				right: width + pageX,
180 				width: width,
181 				height: height
182 			}
183 	    }
184 	    
185 	};

188 	/**
189      * Calculate css left and top to element on context
190      * @private
191      * @function
192      * @name getPosition
193      * @return {Axis Object}
194      * @memberOf ch.Positioner
195      */
196 	var getPosition = function(unitPoints) {		     
197 		// my_x and at_x values together
198 		// cache properties 
199 		var contextLeft = context.left;
200 		var contextTop = context.top;
201 		var contextWidth = context.width;
202 		var contextHeight = context.height;
203 		var elementWidth = element.outerWidth();
204 		var elementHeight = element.outerHeight();
205 		
206 		var xReferences = {
207 			ll: contextLeft,
208 			lr: contextLeft + contextWidth,
209 			rr: contextLeft + contextWidth - elementWidth,
210 			cc: contextLeft + contextWidth/2 - elementWidth/2
211 			// TODO: lc, rl, rc, cl, cr
212 		}
213 		
214 		// my_y and at_y values together
215 		var yReferences = {
216 			// jquery 1.6 do not support offset on IE
217 			tt: contextTop,
218 			tb: contextTop + contextHeight,
219 			bt: contextTop - elementHeight,
220 			mm: contextTop + contextHeight/2 - elementHeight/2
221 			// TODO: tm, bb, bm, mt, mb
222 		}
223 		
224 		var axis = {
225 			left: xReferences[unitPoints.my_x + unitPoints.at_x],
226 			top: yReferences[unitPoints.my_y + unitPoints.at_y]	
227 		} 
228 
229 		return axis;
230 	};
231 	
232     /**
233      * Evaluate viewport spaces and set points
234      * @private
235      * @function
236      * @name calculatePoints
237      * @return {Styles Object}
238      * @memberOf ch.Positioner
239      */
240 	var calculatePoints = function(points, unitPoints){					
241 		// Default styles
242         var styles = getPosition(unitPoints);
243         	styles.direction = classReferences[points];
244 		
245 		// Hold behavior
246 		if (o.hold) return styles;
247 
248         // Check viewport limits	
249 		// Bottom to top
250 		if ( (points == "lt lb") && ((styles.top + parentRelative.top + element.outerHeight()) > viewport.bottom) ) { // Element bottom > Viewport bottom
251 			unitPoints.my_y = "b";
252 			unitPoints.at_y = "t";
253 
254 			//store old styles
255 			stylesBottom = styles;
256 			
257 			// New styles		 
258 			styles = getPosition(unitPoints);
259 			styles.direction = "top";
260 			styles.top -= (2 * offset_top);
261 		
262 			// Top to Bottom - Default again 
263 			if(styles.top + parentRelative.top < viewport.top){
264 				unitPoints.my_y = "t";
265 				unitPoints.at_y = "b";
266 				styles = stylesBottom;
267 				styles.direction = "bottom";
268 			};
269 		};
270 		
271 		// Left to right
272 		if ( (styles.left + parentRelative.left + element.outerWidth()) > viewport.right ) { // Element right > Viewport right
273 			unitPoints.my_x = "r";
274 			unitPoints.at_x = "r";
275 			
276 			//store old styles
277 			stylesLeft = styles;
278 			
279 			// New styles
280 			var current = styles.direction;
281 			styles = getPosition(unitPoints);
282 			styles.direction = current + "-right";						
283 			styles.left -= (2 * offset_left);
284 			if(current == "top") styles.top -= (2 * offset_top);
285 			
286 			// Right to Left - Default again 
287 			if(styles.left < viewport.left){
288 				unitPoints.my_y = "l";
289 				unitPoints.at_y = "l";
290 				styles = stylesLeft;
291 			};
292 		};
293 
294 		return styles;
295 	};

298     /**
299      * Set position to element
300      * @private
301      * @function
302      * @name setPosition
303      * @memberOf ch.Positioner
304      */
305 	var setPosition = function() {
306 		// Separate points config
307         var splitted = o.points.split(" ");
308         
309         var unitPoints = {
310         	my_x: splitted[0].slice(0,1),
311         	my_y: splitted[0].slice(1,2),
312         	at_x: splitted[1].slice(0,1),
313         	at_y: splitted[1].slice(1,2)
314         }
315 
316 		var styles = calculatePoints(o.points, unitPoints);
317 
318 		element
319 			.css({
320 				left: styles.left,
321 				top: styles.top
322 			})
323 			.removeClass( "ch-top ch-left ch-bottom ch-right ch-bottom-right ch-top-right  ch-right-right" )
324 			.addClass( "ch-" + styles.direction );
325 				
326 		if ( context.hasOwnProperty("element") && context.element !== ch.utils.window[0] ){
327 			$(context.element)
328 				.removeClass( "ch-top ch-left ch-bottom ch-right ch-bottom-right ch-top-right ch-right-right" )
329 				.addClass( "ch-" + styles.direction );
330 		};
331 
332 	};

334     /**
335      * Get context element for positioning, if ain't one, select the viewport as context.
336      * @private
337      * @function
338      * @name getContext
339      * @return {Context Object}
340      * @memberOf ch.Positioner
341      */
342 	var getContext = function(){
343 	    
344 	    if (!o.context) {
345 	        return viewport;
346 	    }
347 	     
348         var contextOffset = o.context.offset();
349         
350         context = {
351             element: o.context,
352             top: contextOffset.top + offset_top - parentRelative.top,
353             left: contextOffset.left + offset_left - parentRelative.left,
354             width: o.context.outerWidth(),
355             height: o.context.outerHeight()
356         };
357         
358         return context;	        
359 	    
360 	};
361 	
362     /**
363      * Get offset values from relative parents
364      * @private
365      * @function
366      * @name getParentRelative
367      * @return {Offset Object}
368      * @memberOf ch.Positioner 
369      */
370 	var getParentRelative = function(){
371 		
372 		var relative = {};
373 			relative.left = 0;
374 			relative.top = 0;
375 		
376 		var parent = element.offsetParent();
377 
378 		if ( parent.css("position") == "relative" ) {
379 			
380 			var borderLeft = (parent.outerWidth() - parent.width() - ( parseInt(parent.css("padding-left")) * 2 )) / 2;
381 			
382 			relative = parent.offset();
383 			relative.left -= offset_left - borderLeft;
384 			relative.top -= offset_top;
385 			
386 		};
387 		
388 		return {
389 			left: relative.left,
390 			top: relative.top
391 		};
392 		
393 	};

396 	var scrolled = false;
397 
398 	// Scroll and resize events
399 	// Tested on IE, Magic! no lag!!
400 	ch.utils.window.bind("resize scroll", function() {
401 		scrolled = true;
402 	});
403 	
404 	setInterval(function() {
405 	    if( !scrolled ) return;
406 		scrolled = false;
407 		// Hidden behavior
408 		if( element.css("display") === "none" ) return; 	
409 		initPosition();
410 	}, 250);
411 
412    /**
413     * @ignore
414     */
415     initPosition();
416 	
417 	// Return the reference to the positioned element
418 	return $(element);
419 };
