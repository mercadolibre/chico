  2 /**
  3  * Validate numbers.
  4  * @name Number
  5  * @class Number
  6  * @augments ch.Watcher
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties.
  9  * @return {Chico-UI Object}
 10  * @see ch.Min
 11  * @see ch.Max
 12  * @see ch.Price
 13  * @example
 14  * // Create a number validation
 15  * $("input").number("This field must be a number.");
 16  */
 17 
 18 ch.number = function(conf) {
 19 
 20 /**
 21  *  Constructor
 22  */
 23 
 24 	conf = conf || {};
 25 	
 26     conf.messages = conf.messages || {};
 27 
 28     if ( conf.hasOwnProperty("msg") ) { 
 29 		conf.messages.number = conf.msg;
 30     	conf.msg = null;
 31     	delete conf.msg;
 32     };
 33     
 34     // $.number("message"); support
 35 	if ( !conf.number && !conf.min && !conf.max && !conf.price ) {
 36 		conf.number = true;
 37 	};
 38   
 39 	// Add validation types
 40 	conf.types = "number,min,max,price";
 41     
 42     // Define the conditions of this interface
 43     conf.conditions = [{
 44             name: "number",
 45             patt: /^([0-9\s]+)$/ 
 46     	},{
 47             name: "min",
 48             expr: function(a,b) { return a >= b } 
 49         },{
 50             name: "max",
 51             expr: function(a,b) { return a <= b } 
 52         },{
 53             name: "price",
 54             patt: /^(\d+)[.,]?(\d?\d?)$/ 
 55         }];
 56 
 57 
 58 	return ch.watcher.call(this, conf);
 59     
 60 };

 62 /**
 63  * Validate a number with a minimun value.
 64  * @name Min
 65  * @class Min
 66  * @augments ch.Number
 67  * @augments ch.Watcher
 68  * @memberOf ch
 69  * @param {Number} value Minimun number value.
 70  * @param {String} [message] Validation message.
 71  * @return {Chico-UI Object}
 72  * @see ch.Max
 73  * @see ch.Price
 74  * @example
 75  * $("input").min(10, "Write a number bigger than 10");
 76  */
 77 
 78 ch.min = function(conf) {
 79     
 80     conf = conf || {};
 81 	
 82 	conf.min = conf.value;
 83 	
 84 	conf.value = null;
 85 	
 86 	delete conf.value;
 87 	
 88 	conf.messages = {};
 89 
 90     if ( conf.hasOwnProperty("msg") ) { 
 91 		conf.messages.min = conf.msg;
 92     	conf.msg = null;
 93     	delete conf.msg;
 94     };
 95 
 96 	return ch.number.call(this, conf);
 97     
 98 };
