 52 	/**
 53      * Creates the necesary structure to carousel operation.
 54      * @private
 55      * @function
 56      * @name _createLayout
 57      * @memberOf ch.Carousel
 58      */
 59 	var _createLayout = function(){
 60 
 61 		// Create carousel's content
 62 		that.$content = $("<div>")
 63 			.addClass("ch-carousel-content")
 64 			.append( that.$collection );
 65 
 66 		// Create carousel's mask
 67 		that.$container = $("<div>")
 68 			.addClass("ch-carousel-container")
 69 			.css("height", that.itemSize.height)
 70 			.append( that.$content )
 71 			.appendTo( that.$element );
 72 
 73 		// Visual configuration
 74 		if ( ch.utils.hasOwn(conf, "width") ) { that.$element.css("width", conf.width); };
 75 		if ( ch.utils.hasOwn(conf, "height") ) { that.$element.css("height", conf.height); };
 76 		if ( !conf.fx && ch.features.transition ) { that.$content.addClass("ch-carousel-nofx"); };
 77 
 78 		return;
 79 	},

 81 	/**
 82      * Creates Previous and Next arrows.
 83      * @private
 84      * @function
 85      * @name _createArrows
 86      * @memberOf ch.Carousel
 87      */
 88 	_createArrows = function(){
 89 		
 90 		// Previous arrow
 91 		that.prevArrow = $("<p>")
 92 			.addClass("ch-prev-arrow")
 93 			.append("<span>Previous</span>")
 94 			.bind("click", that.prev);
 95 		
 96 		// Next arrow
 97 		that.nextArrow = $("<p>")
 98 			.addClass("ch-next-arrow")
 99 			.append("<span>Next</span>")
100 			.bind("click", that.next);
101 		
102 		// Continue carousel arrows behavior
103 		if ( !conf.rolling ) { that.prevArrow.addClass("ch-hide") };
104 		
105 		// Append arrows to carousel
106 		that.$element.prepend(that.prevArrow).append(that.nextArrow);
107 		
108 		// Positions arrows vertically inside carousel
109 		var arrowsPosition = (that.$element.outerHeight() - that.nextArrow.outerHeight()) / 2;
110 		$(that.prevArrow).css("top", arrowsPosition);
111 		$(that.nextArrow).css("top", arrowsPosition);
112 
113 		return;
114 	},

116 	/**
117      * Manages arrows turning it on and off when non-continue carousel is on first or last page.
118      * @private
119      * @function
120      * @name _toggleArrows
121      * @param {Number} page Page to be moved
122      * @memberOf ch.Carousel
123      */
124 	_toggleArrows = function(page){
125  		
126  		// Both arrows shown on carousel's middle
127 		if (page > 1 && page < that.pages ){
128 			that.prevArrow.removeClass("ch-hide");
129 			that.nextArrow.removeClass("ch-hide");
130 		} else {
131 			// Previous arrow hidden on first page
132 			if (page == 1) { that.prevArrow.addClass("ch-hide"); that.nextArrow.removeClass("ch-hide"); };
133 			
134 			// Next arrow hidden on last page
135 			if (page == that.pages) { that.prevArrow.removeClass("ch-hide"); that.nextArrow.addClass("ch-hide"); };
136 		};
137 
138 		return;
139 	},

141 	/**
142      * Creates carousel pagination.
143      * @private
144      * @function
145      * @name _createPagination
146      * @memberOf ch.Carousel
147      */
148 	_createPagination = function(){
149 		
150 		// Deletes pagination if already exists
151 		that.$element.find(".ch-carousel-pages").remove();
152 		
153 		// Create an list of elements for new pagination
154 		that.$pagination = $("<ul>").addClass("ch-carousel-pages");
155 
156 		// Create each mini thumbnail
157 		for (var i = 1; i <= that.pages; i += 1) {
158 			// Thumbnail <li>
159 			var thumb = $("<li>").html(i);
160 			
161 			// Mark as actived if thumbnail is the same that current page
162 			if (i == that.currentPage) { thumb.addClass("ch-carousel-pages-on"); };
163 			
164 			// Append thumbnail to list
165 			that.$pagination.append(thumb);
166 		};
167 
168 		// Bind each thumbnail behavior
169 		$.each(that.$pagination.children(), function(i, e){
170 			$(e).bind("click", function(){
171 				that.goTo(i + 1);
172 			});
173 		});
174 		
175 		// Append list to carousel
176 		that.$element.append( that.$pagination );
177 		
178 		// Positions list
179 		that.$pagination.css("left", (that.$element.outerWidth() - that.$pagination.outerWidth()) / 2);
180 		
181 		// Save each generated thumb into an array
182 		_$itemsPagination = that.$pagination.children();
183 			
184 		return;
185 	},

187 	/**
188      * Calculates items amount on each page.
189      * @private
190      * @function
191      * @name _getItemsPerPage
192      * @memberOf ch.Carousel
193      * @returns {Number} Items amount on each page
194      */
195 	_getItemsPerPage = function(){
196 		// Space to be distributed among all items
197 		var _widthDiff = that.$element.outerWidth() - that.itemSize.width;
198 		
199 		// If there are space to be distributed, calculate pages
200 		return (_widthDiff > that.itemSize.width) ? ~~(_widthDiff / that.itemSize.width) : 1;
201 	},
202 	
203 	/**
204      * Calculates total amount of pages.
205      * @private
206      * @function
207      * @name _getPages
208      * @memberOf ch.Carousel
209      * @returns {Number} Total amount of pages
210      */
211 	_getPages = function(){
212 		// (Total amount of items) / (items amount on each page)
213 		return  Math.ceil( that.items.children.length / that.itemsPerPage );
214 	},
215 
216 	/**
217      * Calculates all necesary data to draw carousel correctly.
218      * @private
219      * @function
220      * @name _draw
221      * @memberOf ch.Carousel
222      */
223 	_draw = function(){
224 		
225 		// Reset size of carousel mask
226 		_maskWidth = that.$container.outerWidth();
227 
228 		// Recalculate items amount on each page
229 		that.itemsPerPage = _getItemsPerPage();
230 		
231 		// Recalculate total amount of pages
232 		that.pages = _getPages();
233 		
234 		// Calculate variable margin between each item
235 		var _itemMargin = (( _maskWidth - (that.itemSize.width * that.itemsPerPage) ) / that.itemsPerPage ) / 2;
236 		
237 		// Modify sizes only if new items margin are positive numbers
238 		if (_itemMargin < 0) return;
239 		
240 		// Detach content from DOM for make a few changes
241 		that.$content.detach();
242 		
243 		// Move carousel to first page for reset initial position
244 		that.goTo(1);
245 		
246 		// Sets new margin to each item
247 		$.each(that.items.children, function(i, e){
248 			e.style.marginLeft = e.style.marginRight = _itemMargin + "px";
249 		});
250 		
251 		// Change content size and append it to DOM again
252 		that.$content
253 			.css("width", ((that.itemSize.width + (_itemMargin * 2)) * that.items.size()) )
254 			.appendTo(that.$container);
255 		
256 		// Create pagination if there are more than one page on total amount of pages
257 		if ( conf.pagination && that.pages > 1) { _createPagination(); };
258 		
259 		return;
260 	},

332 	that.goTo = function(page){
333 		
334 		// Validation of page parameter
335 		if (page == that.currentPage || page > that.pages || page < 1 || isNaN(page)) { return that; };
336 		
337 		// Coordinates to next movement
338 		var movement = -(_maskWidth * (page - 1));
339 
340 		// TODO: review this conditional
341 		// Movement with CSS transition
342 		if (conf.fx && ch.features.transition) {
343 			that.$content.css("left", movement);
344 		// Movement with jQuery animate
345 		} else if (conf.fx){
346 			that.$content.animate({ left: movement });
347 		// Movement without transition or jQuery
348 		} else {
349 			that.$content.css("left", movement);
350 		};
351 
352 		// Manage arrows
353 		if (!conf.rolling && conf.arrows) { _toggleArrows(page); };
354 		
355 		// Refresh selected page
356 		that.currentPage = page;
357 		
358 		// TODO: Use toggleClass() instead remove and add.
359 		// Select thumbnail on pagination
360 		if (conf.pagination) {
361 			_$itemsPagination
362 				.removeClass("ch-carousel-pages-on")
363 				.eq(page - 1)
364 				.addClass("ch-carousel-pages-on");
365 		};
366 		
367 		// Movement callback
368 		that.callbacks("onMove");
369 
370  		return that;
371 	};
372

377

382 		that.goTo(that.currentPage + 1);
