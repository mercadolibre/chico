  2 /**
  3  * Carousel is a UI-Component.
  4  * @name Carousel
  5  * @class Carousel
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11  
 12 ch.carousel = function(conf){
 13 	
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.Carousel
 20      */	
 21 	var that = this;
 22 	
 23 	that.$element.addClass('ch-carousel');
 24 	
 25 	if ( conf.height ) that.$element.height( conf.height );
 26 	if ( conf.width ) that.$element.width( conf.width );
 27 	if ( conf.hasOwnProperty("arrows") ) conf.arrows = conf.arrows; else conf.arrows = true;
 28 	
 29 	// UL configuration
 30 	that.$content = that.$element.find('.carousel')	 // TODO: wrappear el contenido para que los botones se posicionen con respecto a su contenedor
 31 		.addClass('ch-carousel-content')
 32 		.wrap($('<div>').addClass('ch-mask'))//gracias al que esta abajo puedo leer el $mask.width()
 33 	
 34 	conf = ch.clon(conf);
 35 	that.conf = conf;
 36 	
 37 /**
 38  *	Inheritance
 39  */
 40 
 41     that = ch.sliders.call(that);
 42     that.parent = ch.clon(that);
 43 
 44 /**
 45  *  Private Members
 46  */
 47  
 48     /**
 49      * Page size.
 50      * @private
 51      * @name page
 52      * @type {Number}
 53      * @memberOf ch.Carousel
 54      */ 
 55 	var page = 1;
 56 	
 57 	// UL Width calculator
 58 	var htmlElementMargin = parseInt( that.$content.children().css("marginLeft") ) * 2;
 59 
 60 	var extraWidth = (ch.utils.html.hasClass("ie6")) ? that.$content.children().outerWidth() : 0;
 61 	var htmlContentWidth = that.$content.children().size() * (that.$content.children().outerWidth() + htmlElementMargin) + extraWidth;
 62 	that.$content.css('width', htmlContentWidth);
 63 	
 64 	// Mask Object and draw function	
 65 	var $mask = that.$element.find('.ch-mask');
 66 	var steps, totalPages, moveTo, margin;
 67 	
 68     /**
 69      * Calculate size of the carousel's mask.
 70      * @private
 71      * @name calculateMask
 72      * @function
 73      * @return {That Object}
 74      * @memberOf ch.Carousel
 75      */ 
 76 	var calculateMask = function(){
 77 		// Steps = (width - marginMask / elementWidth + elementMargin) 70 = total margin (see css)
 78 		steps = ~~( (that.$element.outerWidth() - 70) / (that.$content.children().outerWidth() + htmlElementMargin));
 79 		steps = (steps == 0) ? 1 : steps;
 80 		totalPages = Math.ceil(that.$content.children().size() / steps);
 81 		
 82 		// Move to... (steps in pixels)
 83 		moveTo = (that.$content.children().outerWidth() + htmlElementMargin) * steps;
 84 		// Mask configuration
 85 		margin = ($mask.width()-moveTo) / 2;
 86 		$mask.width( moveTo ).height( conf.height || that.$content.children().outerHeight() );
 87 		
 88 		return that;
 89 	};
 90 
 91 	calculateMask();
 92 
 93     /**
 94      * Create pager functionlity whitin a carousel instance.
 95      * @private
 96      * @name makePager
 97      * @function
 98      * @return {Pager Object}
 99      * @memberOf ch.Carousel
100      */ 
101  	var makePager = function(){
102 		that.$element.find(".ch-pager").remove();
103 			
104 		var pager = $("<ul class=\"ch-pager\">");
105 		var thumbs = [];
106 		
107 		// Create each mini thumb
108 		for(var i = 1, j = totalPages + 1; i < j; i += 1){
109 			if(i == page) thumbs.push("<li class=\"ch-pager-on\">"); else thumbs.push("<li>");
110 			thumbs.push(i);
111 			thumbs.push("</li>");
112 		};
113 		pager.append( thumbs.join("") );
114 		
115 		// Create pager
116 		that.$element.append( pager );
117 		
118 		// Position
119 		var contextWidth = pager.parent().width();
120 		var pagerWidth = pager.outerWidth();
121 		
122 		pager.css('left', (contextWidth - pagerWidth) / 2);
123 
124 		if ( ch.utils.html.hasClass("ie6") ) { pager.css('top', that.$element.height() + 15) };
125 
126 		// Children functionality
127 		pager.children().each(function(i, e){ //TODO: unificar con el for de arriba (pager)
128 			$(e).bind("click", function(){
129 				that.select(i+1);
130 			});
131 		});
132 
133 		return pager;
134 	};
135 	
136 	var resize = false;
137

143     /**
144      * Internal class that handles buttons behavior.
145      * @private
146      * @name buttons
147      * @memberOf ch.Carousel
148      */ 
149  	that.buttons = {
150 		prev: {
151 			//TODO usar positioner cuando esten todos los casos de posicionamiento
152 			$element: $('<p class="ch-prev"><span>Previous</span></p>').bind('click', function(){ that.move("prev", 1) }),
153 			on: function(){ that.buttons.prev.$element.addClass("ch-prev-on") },
154 			off: function(){ that.buttons.prev.$element.removeClass("ch-prev-on") }
155 		},
156 		next: {
157 			//TODO usar positioner cuando esten todos los casos de posicionamiento
158 			$element: $('<p class="ch-next"><span>Next</span></p>').bind('click', function(){ that.move("next", 1) }),
159 			on: function(){ that.buttons.next.$element.addClass("ch-next-on") },
160 			off: function(){ that.buttons.next.$element.removeClass("ch-next-on") }
161 		},
162 		position: function(){
163 			// 50 = button height + "margin"; 10 = bottom position if pager exists },
164 			var newTop = {'top': (that.$element.outerHeight() - 50 + (( conf.pager ) ? 10 : 0)) / 2};
165 				
166 			that.buttons.prev.$element.css(newTop);
167 			that.buttons.next.$element.css(newTop);
168 		}
169 	};
170 	
171 	that.move = function(direction, distance){
172 		var movement;
173 		
174 		switch(direction){
175 			case "prev":
176 				// Validation
177 				if(that.active || (page - distance) <= 0) return;
178 				
179 				// Next move
180 				page -= distance;
181 				
182 				// Css object
183 				movement = that.$content.position().left + (moveTo * distance);
184 				
185 				// Buttons behavior
186 				if ( conf.arrows ) {
187 					if(page == 1) that.buttons.prev.off();
188 					that.buttons.next.on();
189 				};
190 			break;
191 			case "next":
192 				// Validation
193 				if(that.active || (page + distance) > totalPages) return;
194 				
195 				// Next move
196 				page += distance;
197 				
198 				// Css object
199 				movement = that.$content.position().left - (moveTo * distance);
200 				
201 				// Buttons behavior
202 				if ( conf.arrows ) {
203 					if(page == totalPages) that.buttons.next.off();
204 					that.buttons.prev.on();
205 				};
206 			break;
207 		};
208 				
209 		// Status moving
210 		that.active = true;
211 		
212 		// Function executed after movement
213 		var afterMove = function(){
214 			that.active = false;
215 			
216 			// Pager behavior
217 			if (conf.pager) {								
218 				that.pager.children().removeClass("ch-pager-on");
219 				that.pager.children(":nth-child("+page+")").addClass("ch-pager-on");
220 			};
221 
222 			// Callbacks
223 			that.callbacks("onMove");
224 		};
225 		
226 		// Have CSS3 Transitions feature?
227 		if (ch.features.transition) {
228 			
229 			// Css movement
230 			that.$content.css({ left: movement });
231 			
232 			// Callback
233 			afterMove();
234 			
235 		// Ok, let JQuery do the magic...
236 		} else {
237 			
238 			that.$content.animate({ left: movement }, afterMove);
239 		};
240 		
241 		return that;
242 	};
243 	
244 	
245 	that.select = function(pageToGo){
246 		//var itemPage = ~~(item / steps) + 1; // Page of "item"
247 		
248 		// Move right
249 		if(pageToGo > page){
250 			that.move("next", pageToGo - page);
251 		// Move left
252 		}else if(pageToGo < page){
253 	        that.move("prev", page - pageToGo);
254 		};
255 		
256 		if (conf.pager) {
257 			that.pager.children().removeClass("ch-pager-on");
258 			that.pager.children(":nth-child("+page+")").addClass("ch-pager-on");
259 		};
260 			
261 	    return that;
262 	};
263 	
264 	that.redraw = function(){
265 		if (steps > 1){ that.select(1); } //reset the position
266 		
267 		calculateMask();
268 		
269 		that.buttons.position();
270 		
271 		if (conf.pager && totalPages > 1) that.pager = makePager();
272 	};

307      * @public
308      * @name getSteps
309      * @return {Number}

311      */
312 	that["public"].getSteps = function() { return steps; };
313

316      * @public

318      * @return {Number}

321     that["public"].getPage = function() { return page; };
322     
323     /**
324      * Moves the carousel to the defined page.
325      * @public
326      * @name moveTo
327      * @return {Chico-UI Object}
328      * @memberOf ch.Carousel
329      */
330     that["public"].moveTo = function(page) {
331     	that.select(page);
332 
333     	return that["public"];
334     };
335

337      * Moves to the next page.
338      * @public
339      * @name next
340      * @return {Chico-UI Object}
341      * @memberOf ch.Carousel
342      */
343     that["public"].next = function(){
344     	that.move("next", 1);
345 
346     	return that["public"];
347     };
348

350      * Moves to the previous page.
351      * @public
352      * @name prev
353      * @return {Chico-UI Object}
354      * @memberOf ch.Carousel
355      */
356 	that["public"].prev = function(){
357 		that.move("prev", 1);
358 
359 		return that["public"];
360 	};
361 
362     /**
363      * Re-calculate positioning, sizing, paging, and re-draw.
364      * @public
365      * @name redraw
366      * @return {Chico-UI Object}
367      * @memberOf ch.Carousel
368      */	
369 	that["public"].redraw = function(){
370 		that.redraw();
371 		
372 		return that["public"];
373 	};
374 
375 
376 /**
377  *  Default event delegation
378  */
379  	
380 	// UL width configuration
381 	that.$content.css('width', htmlContentWidth);
382 
383 	// Create pager if it was configured
384 	if (conf.pager){
385 		that.$element.addClass("ch-pager-bottom");
386 		that.pager = makePager();	
387 	};
388 	
389 	// Buttons behavior
390 	if ( conf.arrows ){
391 		that.$element.prepend( that.buttons.prev.$element ).append( that.buttons.next.$element ); // Append prev and next buttons
392 		if (htmlContentWidth > $mask.width()) that.buttons.next.on(); // Activate Next button if items amount is over carousel size
393 		that.buttons.position();
394 	};
395 	
396 	
397 	// Elastic behavior    
398     if ( !conf.hasOwnProperty("width") ){
399 		
400 	    ch.utils.window.bind("resize", function() {
401 			resize = true;
402 		});
403 		
404 		setInterval(function() {
405 		    if( !resize ) return;
406 			resize = false;
407 			that.redraw();
408 		}, 250);
409 		
410 	};
411  
412 	return that;
