  2 /**
  3  * Accordion is a UI-Component.
  4  * @name Accordion
  5  * @class Accordion
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11  
 12 ch.accordion = function(conf){
 13 
 14 // Private members
 15 
 16     /**
 17      * Reference to a internal component instance, saves all the information and configuration properties.
 18      * @private
 19      * @name that
 20      * @type {Object}
 21      * @memberOf ch.Accordion
 22      */	
 23 	var that = this;
 24 	that.$element.addClass('ch-accordion');
 25 	conf = ch.clon(conf);
 26 	that.conf = conf;
 27 	
 28     // Inheritance
 29     that = ch.controllers.call(that);
 30     that.parent = ch.clon(that);
 31 
 32     /**
 33      * The component's instance unique identifier.
 34      * @public
 35      * @name uid
 36      * @type {Number}
 37      * @memberOf ch.Accordion
 38      */ 	
 39 	that["public"].uid = that.uid;
 40 	
 41     /**
 42      * The element reference.
 43      * @public
 44      * @name element
 45      * @type {HTMLElement}
 46      * @memberOf ch.Accordion
 47      */
 48 	that["public"].element = that.element;
 49 	
 50     /**
 51      * The component's type.
 52      * @public
 53      * @name type
 54      * @type {String}
 55      * @memberOf ch.Accordion
 56      */
 57 	that["public"].type = that.type;
 58 	
 59     /**
 60      * Select a specific children.
 61      * @public
 62      * @name select
 63      * @function
 64      * @memberOf ch.Accordion
 65      */
 66 	that["public"].select = function(bellows){
 67 		
 68 		if(typeof bellows == "string") {
 69 			var sliced = bellows.split("#");
 70 		
 71 			that.children[ sliced[0] ].select( sliced[1] );
 72 		} else {
 73 			that.children[ bellows ].show();
 74 		};
 75 		
 76 		that.callbacks("onSelect");
 77 		
 78 		return that["public"];
 79 	};	
 80 	
 81 /**
 82  *  Default event delegation
 83  */	
 84     
 85     // Children
 86 	that.$element.children().each(function(i, e){
 87 		
 88 		var $child = $(e).children();
 89 		
 90 		// Link
 91 		if($child.eq(0).prop("tagName") == "A") {
 92 			$(e).addClass("ch-bellows").children().addClass("ch-bellows-trigger");
 93 			that.children.push( $child[0] );
 94 			return;
 95 		};
 96 		
 97 		// Bellows
 98 		var list = {};
 99 			list.uid = that.uid + "#" + i;
100 			list.type = "bellows";
101 			list.element = e;
102 			list.$element = $(e);
103 			
104 		// Selected -> It can be for example "2" or "2#1"
105 		if(conf.hasOwnProperty("selected")) {
106 			list.open = (typeof conf.selected == "number") ? conf.selected == i : (conf.selected.split("#")[0] == i) ? conf.selected.split("#")[1] : false;
107 		} else {
108 			list.open = false;
109 		};
110 			
111 		that.children.push( ch.bellows.call(list, that) );
112 	});
113     
114     
115 	return that;
116 	
117 };

121 /**
122  * Accordion's content container.
123  * @name Bellows
124  * @class Bellows
125  * @augments ch.Navs
126  * @memberOf ch
127  * @param {Configuration Object} conf Object with configuration properties
128  * @return {Chico-UI Object}
129  */
130 ch.bellows = function(controller){
131 
132     /**
133      * Reference to a internal component instance, saves all the information and configuration properties.
134      * @private
135      * @name that
136      * @type {Object}
137      * @memberOf ch.Bellows
138      */ 
139 	var that = this;
140 	
141 	conf = {};
142 	
143 	that.conf = conf;
144 /**
145  *	Inheritance
146  */
147 
148     that = ch.navs.call(that);
149     that.parent = ch.clon(that);
150 	that.controller = controller;
151 	
152 /**
153  *  Protected Members
154  */ 
155 	
156 	that.$container = that.$element.addClass("ch-bellows");
157 	
158 	that.$trigger = that.$container.children(":first");
159 	
160     /**
161      * Component's content.
162      * @public
163      * @name $content
164      * @type {jQuery Object}
165      * @memberOf ch.Bellows
166      */
167 	that.$content = that.$trigger.next();
168     
169     /**
170      * Shows component's content.
171      * @public
172      * @name show
173      * @return {Chico-UI Object}
174      * @memberOf ch.Bellows
175      */
176 	that.show = function(event){
177 
178 		that.prevent(event);
179 
180 		// Toggle
181 		if (that.active) return that.hide(event);
182 
183 		// Accordion behavior (hide last active)
184 		if (!controller.conf.hasOwnProperty("menu")) {
185 
186 			$.each(controller.children, function(i, e){
187 				if (e.tagName == "A") return;
188 				if ( e.hasOwnProperty("active") && e.hasOwnProperty("element") ) {
189 					if (e.active == true && e.element !== that.element) { e.hide(); };
190 				}; 
191 			});
192 			
193 		};
194 		
195 		that.$content.slideDown("fast");
196         
197         that.parent.show(event);
198         
199         return that;
200     };
201     
202     /**
203      * Hides component's content.
204      * @public
205      * @name hide
206      * @return {Chico-UI Object}
207      * @memberOf ch.Bellows
208      */
209     that.hide = function(event){
210     	that.prevent(event);
211     	
212     	// Toggle
213     	if (!that.active) return;
214     	
215     	that.active = false;
216     	
217    		that.parent.hide(event);
218     	that.$content.slideUp("fast");
219 
220         that.$trigger.removeClass("ch-" + that.type + "-on");
221         
222 		if(!ch.utils.html.hasClass("ie6")) that.$content.slideUp("fast");
223 		
224 		that.callbacks("onHide");
225 		
226         return that;
227 	};
228 	
229     /**
230      * Select component's content.
231      * @public
232      * @name select
233      * @return {Chico-UI Object}
234      * @memberOf ch.Bellows
235      */
236 	that.select = function(child) {
237 		that.show();
238 		
239 		// L2 selection
240 		that.$content.find("a").eq( child ).addClass("ch-bellows-on");
241 	};
242 	
243 	
244 /**
245  *  Default event delegation
246  */	 	
247 	
248 	// Closed by default
249 	if(that.open) that.select( parseInt(that.open) ); else that.$content.addClass("ch-hide");
250 	
251 	// Trigger
252 	that.$trigger
253 		.addClass("ch-bellows-trigger")
254 		.bind("click", function(event){ that.show(event) })
255 		.append("<span class=\"ch-arrow\"> »</span>");
256 
257 	// Content
258 	that.$content
259 		.addClass("ch-bellows-content")
260 		.bind("click", function(event){ event.stopPropagation(); });
261 	
262 	
263 	return that;
264 };
