  2 /**
  3  * Is a contextual floated UI-Object.
  4  * @name Layer
  5  * @class Layer
  6  * @augments ch.Floats
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  * @see ch.Tooltip
 11  * @see ch.Modal
 12  * @example
 13  * // Create a simple contextual layer
 14  * $("element").layer("<p>Content.</p>");
 15  */ 
 16 
 17 ch.layer = function(conf) {
 18     
 19     /**
 20      * Reference to a internal component instance, saves all the information and configuration properties.
 21      * @private
 22      * @name that
 23      * @type {Object}
 24      * @memberOf ch.Layer
 25      */ 
 26 	var that = this;
 27 	
 28 	conf = ch.clon(conf);
 29 	conf.cone = true;
 30 	conf.classes = "box";
 31 	conf.position = {};
 32 	conf.position.context = that.$element;
 33 	conf.position.offset = conf.offset || "0 10";
 34 	conf.position.points = conf.points || "lt lb";
 35 
 36 	that.conf = conf;
 37 
 38 /**
 39  *	Inheritance
 40  *		
 41  */
 42 
 43     that = ch.floats.call(that);
 44     that.parent = ch.clon(that);
 45     
 46 /**
 47  *  Private Members
 48  */
 49  
 50     /**
 51      * Delay time to show component's contents.
 52      * @private
 53      * @name showTime
 54      * @type {Number}
 55      * @default 400
 56      * @memberOf ch.Layer
 57      */ 
 58     var showTime = conf.showTime || 400;
 59     /**
 60      * Delay time to hide component's contents.
 61      * @private
 62      * @name hideTime
 63      * @type {Number}
 64      * @default 400
 65      * @memberOf ch.Layer
 66      */ 
 67     var hideTime = conf.hideTime || 400;
 68 
 69     /**
 70      * Show timer instance.
 71      * @private
 72      * @name st
 73      * @type {Timer}
 74      * @memberOf ch.Layer
 75      */ 
 76 	var st;
 77 	/**
 78      * Hide timer instance.
 79      * @private
 80      * @name ht
 81      * @type {Timer}
 82      * @memberOf ch.Layer
 83      */ 
 84 	var ht;
 85     /**
 86      * Starts show timer.
 87      * @private
 88      * @name showTimer
 89      * @function
 90      * @memberOf ch.Layer
 91      */ 
 92 	var showTimer = function(){ st = setTimeout(that.show, showTime) };
 93     /**
 94      * Starts hide timer.
 95      * @private
 96      * @name hideTimer
 97      * @function
 98      * @memberOf ch.Layer
 99      */ 
100 	var hideTimer = function(){ ht = setTimeout(that.hide, hideTime) };
101     /**
102      * Clear all timers.
103      * @private
104      * @name clearTimers
105      * @function
106      * @memberOf ch.Layer
107      */ 
108 	var clearTimers = function(){ clearTimeout(st); clearTimeout(ht); };
109 
110 /**
111  *  Protected Members
112  */
113 	
114 	that.$trigger = that.$element;
115 	
116 	that.show = function(event) {
117 	
118 		// Reset all layers
119 		$.each(ch.instances.layer, function(i, e){ e.hide(); });
120 		//conf.position.context = that.$element;
121 		that.parent.show(event);
122 
123 		that.$container.bind('click', function(event){ event.stopPropagation() });
124         
125         // Click
126         if (conf.event == "click") {
127 			conf.close = true;
128             // Document events
129             $(document).one('click', that.hide);
130             
131         // Hover
132         } else {      	
133         	clearTimers();    
134         	that.$container
135         		.one("mouseenter", clearTimers)
136         		.one("mouseleave", function(event){
137 					var target = event.srcElement || event.target;
138 					var relatedTarget = event.relatedTarget || event.toElement;
139 					var relatedParent = relatedTarget.parentNode;
140 					if ( target === relatedTarget || relatedParent === null || target.nodeName === "SELECT" ) return;
141 					hideTimer();
142         		});
143         };
144         
145         return that;
146     };
147 
148 /**
149  *  Public Members
150  */
151  
152     /**
153      * The component's instance unique identifier.
154      * @public
155      * @name uid
156      * @type {Number}
157      * @memberOf ch.Layer
158      */
159    	that["public"].uid = that.uid;
160     /**
161      * The element reference.
162      * @public
163      * @name element
164      * @type {HTMLElement}
165      * @memberOf ch.Layer
166      */
167 	that["public"].element = that.element;
168     /**
169      * The component's type.
170      * @public
171      * @name type
172      * @type {String}
173      * @memberOf ch.Layer
174      */
175 	that["public"].type = that.type;
176     /**
177      * The component's content.
178      * @public
179      * @name content
180      * @type {String}
181      * @memberOf ch.Layer
182      */
183 	that["public"].content = that.content;
184     /**
185      * Shows component's content.
186      * @public
187      * @name show
188      * @function
189      * @return {Chico-UI Object}
190      * @memberOf ch.Layer
191      */
192 	that["public"].show = function(){
193 		that.show();
194 		
195 		return that["public"];
196 	};
197     /**
198      * Hides component's content.
199      * @public
200      * @name hide
201      * @function
202      * @return {Chico-UI Object}
203      * @memberOf ch.Layer
204      */	
205 	that["public"].hide = function(){
206 		that.hide();
207 		
208 		return that["public"];
209 	};
210     /**
211      * Positioning configuration.
212      * @public
213      * @name position
214      * @memberOf ch.Layer
215      * @example
216      * // Change layer's position.
217      * $('input').layer("content").position({ 
218      *    offset: "0 10",
219      *    points: "lt lb"
220      * });
221      */
222 	that["public"].position = that.position;
223 	
224 /**
225  *  Default event delegation
226  */
227 	// Click
228 	if(conf.event === 'click') {
229 		// Local configuration
230 		conf.closeButton = true;
231 
232 		// Trigger events
233 		that.$trigger
234 			.css('cursor', 'pointer')
235 			.bind('click', that.show);
236 
237 	// Hover
238 	} else {
239 		// Trigger events
240 		that.$trigger
241 			.css('cursor', 'default')
242 			.bind('mouseenter', that.show)
243 			.bind('mouseleave', hideTimer);
244 	};
245 
246     // Fix: change layout problem
247     $("body").bind(ch.events.CHANGE_LAYOUT, function(){ that.position("refresh") });
248  
249 
250 	return that;
251 
252 };
