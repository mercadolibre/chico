  2 /**
  3  * Forms is a Controller of DOM's HTMLFormElement.
  4  * @name Form
  5  * @class Form
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11 
 12 ch.form = function(conf) {
 13 
 14 /**
 15  *  Validation
 16  */
 17 	// Are there action and submit type?
 18 	if ( this.$element.find(":submit").length == 0 || this.$element.attr("action") == "" ){ 
 19 		alert("Form fatal error: The <input type=submit> is missing, or need to define a action attribute on the form tag.");
 20 		return;
 21 	};
 22 
 23 	// Is there form in map instances?	
 24 	if ( ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0 ){
 25 		for(var i = 0, j = ch.instances.form.length; i < j; i++){
 26 			if(ch.instances.form[i].element === this.element){
 27 				return { 
 28 	                exists: true, 
 29 	                object: ch.instances.form[i]
 30 	            };
 31 			};
 32 		};
 33 	};
 34 
 35     /**
 36      * Reference to a internal component instance, saves all the information and configuration properties.
 37      * @private
 38      * @name that
 39      * @type {Object}
 40      * @memberOf ch.Form
 41      */ 
 42 	var that = this;
 43 	
 44 	conf = ch.clon(conf);
 45 	// Create the Messages for General Error
 46 	if ( !ch.utils.hasOwn(conf, "messages") ) conf.messages = {};
 47 	conf.messages["general"] = conf.msg || conf.messages["general"]  || "Check for errors.";	
 48 	
 49 	// Disable HTML5 browser-native validations
 50 	that.$element.attr("novalidate", "novalidate");
 51 	// Grab submit button
 52 	that.$submit = that.$element.find("input:submit");
 53 	
 54 	that.conf = conf;
 55 
 56 /**
 57  *  Inheritance
 58  */
 59 
 60     that = ch.controllers.call(that);
 61     that.parent = ch.clon(that);
 62 	
 63 	
 64 /**
 65  *  Private Members
 66  */
 67  
 68     /**
 69      * A Boolean property that indicates is exists errors in the form.
 70      * @private
 71      * @name status
 72      * @type {Boolean}
 73      * @memberOf ch.Form
 74      */ 
 75 	var status = true;
 76 	
 77     /**
 78      * HTML snippet to show the general error on top of the form.
 79      * @private
 80      * @name $error
 81      * @type {jQuery Object}
 82      * @memberOf ch.Form
 83      */ 
 84 	var $error = $("<p class=\"ch-validator\"><span class=\"ico error\">Error: </span>" + conf.messages["general"] + "</p>");
 85 	
 86     /**
 87      * Inserts the general error snippet into the HTML. This implies a change in the document's flow, so it will trigger the CHANGE_LAYOUT Event.
 88      * @private
 89      * @function
 90      * @name createError
 91      * @memberOf ch.Form
 92      * @see ch.events.CHANGE_LAYOUT
 93      */ 
 94 	var createError = function(){ 
 95 		that.$element.before( $error );		
 96 		$("body").trigger(ch.events.CHANGE_LAYOUT);
 97 	};
 98 
 99     /**
100      * Removes the general error snippet from the HTML. This implies a change in the document's flow, so it will trigger the CHANGE_LAYOUT Event.
101      * @private
102      * @function
103      * @name removeError
104      * @memberOf ch.Form
105      * @see ch.events.CHANGE_LAYOUT
106      */ 
107  	var removeError = function(){
108 		$error.detach();
109 		$("body").trigger(ch.events.CHANGE_LAYOUT);
110 	};
111 
112     /**
113      * Iterates all the Watchers defined as children, for each one of them will check it's active property and returns when finds an error.
114      */
115 	var checkStatus = function(){
116 		// Check status of my childrens
117 		for(var i = 0, j = that.children.length; i < j; i ++){
118 			// Status error (cut the flow)
119 			if( that.children[i].active() ){
120 				if ( !status ) removeError();			
121 				createError();
122 				status = false;
123 				// Issue UI-332: On validation must focus the first field with errors.
124 				// Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
125 				that.children[i].element.focus();
126 				return;
127 			};
128 		};
129 
130 		// Status OK (with previous error)
131 		if ( !status ) {
132 			removeError();
133 			status = true;
134 		};
135 
136 	};
137 	
138     /**
139      * Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
140      */
141 	var validate = function(){
142 
143         that.callbacks("beforeValidate");
144 		
145 		// Status OK (with previous error)
146 		if ( !status ) {
147 			removeError();
148 			status = true;
149 		};
150 		
151         var i = 0, j = that.children.length, toFocus, childrenError = [];
152 		// Shoot validations
153 		for ( i; i < j; i++ ) {
154 		    var child = that.children[i];
155 			// Validate
156 			if ( !child.active() ) {
157                 child.validate();
158             } 
159             // Save children with errors
160             if ( child.active() ) {
161                 childrenError.push( child );
162             }
163 		};
164         
165         // Is there's an error
166         if ( childrenError.length > 0 ) {
167             if ( !status ) {
168                 removeError();
169             }
170             createError();
171             status = false;
172             // Issue UI-332: On validation must focus the first field with errors.
173             // Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
174             childrenError[0].element.focus();
175         }
176 		
177 		status ? that.callbacks("onValidate") : that.callbacks("onError");  
178 
179         that.callbacks("afterValidate");
180         
181 		return that;
182 	};
183 
184     /**
185      * This methods triggers the 'beforSubmit' callback, then will execute validate() method, 
186      * and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
187      */
188 	var submit = function(event){
189 
190         that.prevent(event);
191         
192         that.callbacks("beforeSubmit");
193 
194 		validate(); // Validate start
195 		
196 		if ( status ){ // Status OK
197 			if ( !ch.utils.hasOwn(conf, "onSubmit") ) {
198                 // To fix Webflow dumb navigation,
199                 // Clone the submit button into a hidden field
200     			var hidden = that.$submit.clone();
201     			    hidden.attr("type","hidden");
202 				    hidden.prependTo(that.element);    
203 
204 			    // And send the form
205 				that.element.submit(event);
206 
207 			}else{
208 				that.callbacks("onSubmit");
209 			};
210 		};		
211 
212         that.callbacks("afterSubmit");
213 
214         // re-asign submit event   
215     	that.$element.one("submit", submit);
216         
217 		return that;
218 	};
219 
220     /**
221      * Use this method to clear al validations.
222      */
223 	var clear = function(event){		
224 		that.prevent(event);		
225 		removeError();	
226 		for(var i = 0, j = that.children.length; i < j; i ++) that.children[i].reset(); // Reset helpers		
227 		
228 		that.callbacks("onClear");
229 		
230 		return that;
231 	};
232 
233     /**
234      * Use this method to reset the form's input elements.
235      */	
236 	var reset = function(event){
237 		clear();
238 		that.element.reset(); // Reset html form native
239 		that.callbacks("onReset");
240 		
241 		return that;
242 	};
243

253      * @memberOf ch.Form

255 	that["public"].uid = that.uid;

257      * The element reference.
258      * @public
259      * @name element
260      * @type {HTMLElement}
261      * @memberOf ch.Form
262      */
263 	that["public"].element = that.element;
264     /**
265      * The component's type.
266      * @public
267      * @name type
268      * @type {String}
269      * @memberOf ch.Form
270      */
271 	that["public"].type = that.type;
272     /**
273      * Watcher instances associated to this controller.
274      * @public
275      * @name children
276      * @type {Collection}
277      * @memberOf ch.Form
278      */
279 	that["public"].children = that.children;
280     /**
281      * Collection of messages defined.
282      * @public
283      * @name messages
284      * @type {String}
285      * @memberOf ch.Form
286      */
287 	that["public"].messages = conf.messages;
288     /**
289      * Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
290      * @function
291      * @name validate
292      * @memberOf ch.Form
293      * @returns {Chico-UI Object}
294      */
295 	that["public"].validate = function() { 
296 		validate(); 
297 		
298 		return that["public"]; 
299 	};
300 	
301     /**
302      * This methods triggers the 'beforSubmit' callback, then will execute validate() method, and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
303      * @function
304      * @name submit
305      * @memberOf ch.Form
306      * @returns {Chico-UI Object}
307      */
308 	that["public"].submit = function() { 
309 		submit(); 
310 		
311 		return that["public"]; 
312 	};
313 	
314     /**
315      * Iterates all the Watchers defined as children, for each one of them will check it's active property and returns when finds an error.
316      * @function
317      * @name checkStatus
318      * @memberOf ch.Form
319      * @see ch.watcher.active
320      */
321 	that["public"].checkStatus = function() { 
322 		checkStatus(); 
323 		
324 		return that["public"]; 
325 	};
326 
327     /**
328      * Return the status value.
329      * @function
330      * @name getStatus
331      * @memberOf ch.Form
332      * @returns {Chico-UI Object}
333      */ 	
334 	that["public"].getStatus = function(){
335 		return status;	
336 	};
337 
338     /**
339      * Use this method to clear al validations.
340      * @function
341      * @name clear
342      * @memberOf ch.Form
343      * @returns {Chico-UI Object}
344      */ 
345 	that["public"].clear = function() { 
346 		clear(); 
347 		
348 		return that["public"]; 
349 	};
350     /**
351      * Use this method to clear al validations.
352      * @function
353      * @name reset
354      * @memberOf ch.Form
355      * @returns {Chico-UI Object}
356      */ 
357 	that["public"].reset = function() { 
358 		reset(); 
359 		
360 		return that["public"]; 
361 	};
362 
363 
364 /**
365  *  Default event delegation
366  */	
367 
368 	// patch exists because the components need a trigger
369 	if (ch.utils.hasOwn(conf, "onSubmit")) {
370 		that.$element.bind('submit', function(event){ that.prevent(event); });
371 		// Delete all click handlers asociated to submit button >NATAN: Why?
372 			//Because if you want do something on submit, you need that the trigger (submit button) 
373 			//don't have events associates. You can add funcionality on onSubmit callback
374 		that.$element.find(":submit").unbind('click');
375 	};
376 
377 	// Bind the submit
378 	that.$element.one("submit", submit);
379 	
380 	// Bind the reset
381 	that.$element.find(":reset, .resetForm").bind("click", function(event){ reset(event); });
382 
383 	return that;
384 };
