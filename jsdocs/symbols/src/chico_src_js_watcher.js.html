 40     /**
 41      * Reference to a ch.form controller. If there isn't any, the Watcher instance will create one.
 42      * @private
 43      * @name controller
 44      * @type {Chico-UI Object}
 45      * @memberOf ch.Watcher
 46      */
 47 	var controller = (function() {
 48 		if ( ch.instances.hasOwnProperty("form") && ch.instances.form.length > 0 ) {	
 49 		  var i = 0, j = ch.instances.form.length; 
 50 		  for (i; i < j; i ++) {
 51 				if (ch.instances.form[i].element === that.$element.parents("form")[0]) {
 52 					return ch.instances.form[i]; // Get my parent
 53 				};
 54 			};
 55 		} else {
 56 			that.$element.parents("form").form();
 57 			var last = (ch.instances.form.length - 1);
 58 			return ch.instances.form[last]; // Set my parent
 59 		};
 60 	})();

 63     /**
 64      * Search for instances of Watchers with the same trigger, and then merge it's properties with it.
 65      * @private
 66      * @function
 67      * @name checkInstance
 68      * @return {Instance Object}
 69      * @memberOf ch.Watcher
 70      */	
 71 	var checkInstance = function() {
 72         var instance = ch.instances.watcher;
 73         if ( instance && instance.length > 0 ) {
 74 			for (var i = 0, j = instance.length; i < j; i ++) {
 75                 if (instance[i].element !== that.element) continue;
 76         	    // Merge Conditions        	    
 77                 $.merge(instance[i].conditions, that.conditions);
 78 				return { 
 79 				    exists: true, 
 80 				    object: instance[i] 
 81 			    };
 82 			    
 83             };
 84         };
 85     };

111     /**
112      * Reference is used to assign a context to the positioning preferences.
113      */
114  	that.reference = (function() {
115         var reference;
116         // CHECKBOX, RADIO
117         if ( that.$element.hasClass("options") ) {
118             // Helper reference from will be fired
119             // H4
120             if ( that.$element.find('h4').length > 0 ) {
121                 var h4 = that.$element.find('h4'); // Find h4
122                     h4.wrapInner('<span>'); // Wrap content with inline element
123                 reference = h4.children(); // Inline element in h4 like helper reference	
124             // Legend
125             } else if ( that.$element.prev().prop('tagName') == 'LEGEND' ) {
126                 reference = that.$element.prev(); // Legend like helper reference
127             }
128         // INPUT, SELECT, TEXTAREA
129         } else {
130             reference = that.$element;
131         }
132         return reference;
133     })();

135 	/**
136      * Process conditions and creates a map with all configured conditions, it's messages and validations.
137      */
138     that.conditions = (function(){
139         var c = []; // temp collection
140         var i = 0;  // iteration
141         var t = conf.conditions.length;
142         for ( i; i < t; i++ ) {
143 
144             /**
145              * Process conditions to find out which should be configured.
146              * Add validations and messages to conditions object.
147              */
148             var condition = conf.conditions[i];
149             
150             // If condition exists in the Configuration Object
151             if ( conf[condition.name] ) {
152                 
153                 // Sabe the value
154                 condition.value = conf[condition.name];
155                 
156                 // If there is a message defined for that condition
157                 if ( conf.messages[condition.name] ) {
158                     condition.message = conf.messages[condition.name];
159                 }
160                 
161                 // Push it to the new conditions collection
162                 c.push(condition);
163             }
164         }
165         
166         // return all the configured conditions
167         return c;
168         
169     })(); // Love this ;)

171     /**
172      * Return true is a required conditions is found on the condition collection.
173      * @private
174      * @function
175      * @name isRequired
176      * @memberOf ch.Watcher
177      */
178     that.isRequired = function(){
179         var t = that.conditions.length;
180         while ( t-- ) {   
181             var condition = that.conditions[t];
182             if ( condition.name === "required" && condition.value ) {
183                 return true;
184             }
185         }
186         return false;
187     };

189     /**
190      * Helper is a UI Component that shows the messages of active validations.
191      * @name helper
192      * @type {ch.Helper}
193      * @memberOf ch.Watcher
194      * @see ch.Helper
195      */
196     var helper = {};
197         helper.uid = that.uid + "#0";
198 		helper.type = "helper";
199 		helper.element = that.element;
200 		helper.$element = that.$element;
201

204     /**
205      * Validate executes all configured validations.
206      */
207  	that.validate = function(event) {	
208 		
209 		// Pre-validation: Don't validate disabled or not required & empty elements
210 		if ( that.$element.attr('disabled') ) { return; }
211 
212         var isRequired = that.isRequired()
213 
214         // Avoid fields that aren't required when they are empty or de-activated
215 		if ( !isRequired && that.isEmpty() && that.active === false) { return; }
216         
217         if ( that.enabled && ( that.active === false || !that.isEmpty() || isRequired ) ) {
218 	
219 			that.callbacks('beforeValidate');
220 
221             var t = that.conditions.length;
222             var value = that.$element.val();
223             var gotError = false;
224             
225             while ( t-- ) {
226                 
227             	var condition = that.conditions[t];
228         
229             	if ( that.isRequired() ) {
230                     gotError = that.isEmpty();
231             	}
232             	
233                 if ( condition.patt ) {
234                     gotError = !condition.patt.test(value);
235                 }
236                 
237                 if ( condition.expr ) {
238                     gotError = !condition.expr( value, condition.value );
239                 }
240 
241                 if ( condition.func) {
242                     // Call validation function with 'this' as scope.
243                     gotError = !condition.func.call(that["public"], value); 
244                 }
245                 
246                 if ( gotError ) {
247 
248                     that.callbacks('onError');
249 		
250                 	// Field error style
251                 	that.$element.addClass("error");
252                 
253                 	// Show helper with message
254                 	var text = ( condition.message ) ? condition.message : 
255                 		(controller.hasOwnProperty("messages")) ? controller.messages[condition.name] :
256                 		undefined;
257                 
258                 	that.helper.show( text );
259                 
260                 	that.active = true;
261                 
262                 	var event = (that.tag == 'OPTIONS' || that.tag == 'SELECT') ? "change" : "blur";
263                 
264                 	that.$element.one(event, function(event){ that.validate(event); }); // Add blur or change event only one time
265                 
266                     return;
267                 }
268             
269             } // End for each validation
270             
271 		} // End if Enabled
272 		
273 		// Status OK (with previous error)
274 		if ( that.active || !that.enabled ) {
275 		    // Remove field error style
276 			that.$element.removeClass("error"); 
277             // Hide helper  
278 			that.helper.hide();
279 			// Public status OK
280 			//that.publish.status = that.status =  conf.status = true; // Status OK
281 			that.active = false;
282 			
283 			// If has an error, but complete the field and submit witout trigger blur event
284 			if (event) {
285 				var originalTarget = event.originalEvent.explicitOriginalTarget || document.activeElement; // Moderns Browsers || IE
286 				if (originalTarget.type == "submit") { controller.submit(); };
287 			};
288 			
289 			controller.checkStatus();
290 		};
291         
292         that.callbacks('afterValidate');
293         
294         return that;
295 	};

297     /**
298      * Reset all active validations messages.
299      */
300  	that.reset = function() {
301 		//that.publish.status = that.status = conf.status = true; // Public status OK
302 		that.active = false;
303 		that.$element.removeClass("error");
304 		that.helper.hide(); // Hide helper
305 		that.$element.unbind("blur change", that.validate); // Remove blur and change event
306 		
307 		that.callbacks("onReset");
308 		
309 		return that;
310 	};

312     /**
313      * isEmpty determine if the field has no value selected.
314      */	
315      that.isEmpty = function() {
316 		that.tag = ( that.$element.hasClass("options")) ? "OPTIONS" : that.element.tagName;
317 		switch (that.tag) {
318 			case 'OPTIONS':
319 				return that.$element.find('input:checked').length === 0;
320 			break;
321 			
322 			case 'SELECT':
323 			    var val = that.$element.val();
324 				return parseInt(val) === -1 || val === null;
325 			break;
326 			
327 			case 'INPUT':
328 			case 'TEXTAREA':
329 				return $.trim( that.$element.val() ).length === 0;
330 			break;
331 		};
332 				
333 	};
