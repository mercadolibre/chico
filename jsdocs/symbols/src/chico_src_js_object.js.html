  2 /**
  3  * Object represent the abstract class of all UI Objects.
  4  * @abstract
  5  * @name Object
  6  * @class Object 
  7  * @memberOf ch
  8  * @requires ch.Get
  9  * @requires ch.Factory
 10  * @see ch.Controllers
 11  * @see ch.Floats
 12  * @see ch.Navs
 13  * @see ch.Watcher
 14  */
 15  
 16 ch.object = function(){
 17 	
 18     /**
 19      * Reference to a internal component instance, saves all the information and configuration properties.
 20      * @private
 21      * @name that
 22      * @type {Object}
 23      * @memberOf ch.Object
 24      */ 
 25 	var that = this;	
 26 	var conf = that.conf;
 27 
 28 /**
 29  *  Public Members
 30  */
 31 	that.prevent = function(event) {
 32 		if (event && typeof event == "object") {
 33 		    event.preventDefault();
 34 			event.stopPropagation();
 35 		};
 36 		
 37 		return that;
 38 	};
 39 	
 40 	//TODO: Analizar si unificar that.content (get and set) con that.loadContent(load).
 41 	that.content = function(content){
 42 		
 43 		if ( content === undefined ) {
 44 		
 45 			var content = conf.content || conf.msg;
 46 			return (content) ? (( ch.utils.isSelector(content) ) ? $(content) : content) : ((conf.ajax === true) ? (that.$trigger.attr('href') || that.$trigger.parents('form').attr('action')) : conf.ajax );
 47 		
 48 		} else {
 49 			
 50 			//TODO: We have to cache the content if it's the same
 51 			conf.cache = false;
 52 			
 53 			if ( ch.utils.isUrl(content) ) {
 54 				conf.content = undefined;
 55 				conf.ajax = content;
 56 			} else {
 57 				conf.ajax = false;
 58 				conf.content = content;
 59 			};
 60 	
 61 			if ( that.active ) {
 62 				that.$content.html(that.loadContent());
 63 				return that.position("refresh");
 64 			};
 65 
 66 			if (that.active) that.position("refresh");
 67 			return that["public"];
 68 		};
 69 		
 70 	};
 71 
 72 	that.loadContent = function() {
 73 		// TODO: Properties validation
 74 		//if( self.ajax && (self.content || self.msg) ) { alert('CH: "Ajax" and "Content" can\'t live together.'); return; };
 75 		
 76 		if( conf.ajax === true){
 77 			
 78 			// Load URL from href or form action
 79 			conf.ajaxUrl = that.$element.attr('href') || that.$element.parents('form').attr('action');
 80 			
 81 			// Ajax parameters
 82 			conf.ajaxParams = 'x=x'; // TODO refactor con el header de ajax
 83 			
 84 			// If trigger is a form button...
 85 			if(that.$element.attr('type') == 'submit'){
 86 				conf.ajaxType = that.$element.parents('form').attr('method') || 'GET';
 87 				var serialized = that.$element.parents('form').serialize();
 88 				conf.ajaxParams = conf.ajaxParams + ((serialized != '') ? '&' + serialized : '');
 89 			};
 90 
 91 			// Returns ajax results
 92 
 93 			ch.get({method:"content", that:that});
 94 			
 95 			return '<div class="loading"></div>';
 96 			
 97 		} else if ( conf.ajax || (conf.msg && ch.utils.isUrl(conf.msg)) ){
 98 			// Set url
 99 			conf.ajaxUrl = conf.ajax || conf.msg;
100 
101 			// Ajax parameters
102 			conf.ajaxParams = 'x=x'; // TODO refactor con el header de ajax
103 
104 			// Returns ajax results
105 
106 			ch.get({method:"content", that:that});
107 			return '<div class="loading"></div>';
108 			
109 		} else {
110 			var content = conf.content || conf.msg;		
111 			return ( ch.utils.isSelector(content) ) ? $(content).detach().clone().removeClass("ch-hide").show() : content;			
112 		};
113 
114 	};
115 
116 	that.callbacks = function(when){
117 		if( conf.hasOwnProperty(when) ) {
118 			var context = ( that.controller ) ? that.controller["public"] : that["public"];
119 			return conf[when].call( context );
120 		};
121 	};
122 	
123 	that.position = function(o){
124 	
125 		switch(typeof o) {
126 		 
127 			case "object":
128 				conf.position.context = o.context || conf.position.context;
129 				conf.position.points = o.points || conf.position.points;
130 				conf.position.offset = o.offset || conf.position.offset;				
131 				conf.position.fixed = o.fixed || conf.position.fixed;
132 			
133 				ch.positioner(conf.position);
134 				return that["public"];
135 				break;
136 		
137 			case "string":
138 				if(o != "refresh"){
139 					alert("ChicoUI error: position() expected to find \"refresh\" parameter.");
140 				};
141 
142 				ch.positioner(conf.position);
143 				return that["public"];   			
144 				break;
145 		
146 			case "undefined":
147 				return conf.position;
148 			    break;
149 		};
150 		
151 	};
152 	
153 
154 	 
155  	that["public"] = {};
156 	
157 	return that;
158 };
