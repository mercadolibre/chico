1 /**
  2  * Viewer UI-Component for images, videos and maps.
  3  * @name Viewer
  4  * @class Viewer
  5  * @augments ch.Controllers
  6  * @requires ch.Carousel
  7  * @requires ch.Zoom
  8  * @memberOf ch
  9  * @param {Configuration Object} conf Object with configuration properties
 10  * @returns {Chico-UI Object}
 11  */
 12 
 13 ch.viewer = function(conf){
 14 
 15     /**
 16      * Reference to a internal component instance, saves all the information and configuration properties.
 17      * @private
 18      * @name that
 19      * @type {Object}
 20      * @memberOf ch.Viewer
 21      */
 22  	var that = this;
 23 	conf = ch.clon(conf);
 24 	that.conf = conf;
 25 	
 26 /**
 27  *  Inheritance
 28  */
 29 
 30     that = ch.controllers.call(that);
 31     that.parent = ch.clon(that);
 32 	
 33 /**
 34  *  Private Members
 35  */
 36  
 37     /**
 38      * Reference to the viewer's visual object.
 39      * @private
 40      * @name $viewer
 41      * @type {jQuery Object}
 42      * @memberOf ch.Viewer
 43      */
 44 	var $viewer = that.$element.addClass("ch-viewer");
 45 	conf.width = $viewer.outerWidth();
 46 	conf.height = $viewer.outerHeight();
 47 
 48     /**
 49      * Reference to the viewer's internal content.
 50      * @private
 51      * @name $content
 52      * @type {jQuery Object}
 53      * @memberOf ch.Viewer
 54      */
 55 	var $content = $viewer.children().addClass("ch-viewer-content carousel");
 56 
 57     /**
 58      * Reference to the viewer's display element.
 59      * @private
 60      * @name $display
 61      * @type {jQuery Object}
 62      * @memberOf ch.Viewer
 63      */
 64 	var $display = $("<div>")
 65 		.addClass("ch-viewer-display")
 66 		.append( $content )
 67 		.appendTo( $viewer )
 68 		.carousel({
 69 			width: conf.width,
 70 			arrows: false,
 71 			onMove: function(){
 72 				var carousel = this;
 73 				var page = carousel.getPage();
 74 				that.move(page);
 75 
 76 				// Resize display
 77 				var currentHeight = $(itemsChildren[page - 1]).height();
 78 				$viewer.find(".ch-mask").eq(0).height(currentHeight);
 79 			}
 80 		})
 81 
 82     /**
 83      * Collection of viewer's children.
 84      * @private
 85      * @name items
 86      * @type {Collection}
 87      * @memberOf ch.Viewer
 88      */
 89 	var items = $content.children();
 90     /**
 91      * Amount of children.
 92      * @private
 93      * @name itemsAmount
 94      * @type {Number}
 95      * @memberOf ch.Viewer
 96      */
 97 	var itemsAmount = items.length;
 98     /**
 99      * Collection of anchors finded on items collection.
100      * @private
101      * @name itemsAnchor
102      * @type {Collection}
103      * @memberOf ch.Viewer
104      */
105 	var itemsAnchor = items.children("a");
106     /**
107      * Collection of references to HTMLIMGElements or HTMLObjectElements.
108      * @private
109      * @name itemsChildren
110      * @type {Object}
111      * @memberOf ch.Viewer
112      */
113 	var itemsChildren = items.find("img, object");
114 	
115 	/**
116 	 * 	Zoom
117 	 */
118 	if( ch.utils.hasOwn(ch, "zoom") ) {
119 		var zoomChildren = [];
120 	
121 		$.each(itemsAnchor, function(i, e){
122 			
123 			var component = {
124 				uid: that.uid + "#" + i,
125 				type: "zoom",
126 				element: e,
127 				$element: $(e)
128 			};
129 			
130 			var config = {
131 	    		context: $viewer,
132 	    		onShow: function(){
133 	    			var rest = (ch.utils.body.outerWidth() - $viewer.outerWidth());
134 	    			var zoomDisplayWidth = (conf.width < rest)? conf.width :	(rest - 65 );
135 	    			this.width( zoomDisplayWidth );
136 	    			this.height( $viewer.height() );
137 	    		}
138 	    	};
139 			
140 			zoomChildren.push( ch.zoom.call(component, config) );
141 		});
142 		
143 		that.children.push( zoomChildren );
144 	};
145 	
146 	/**
147 	 * 	Thumbnails
148 	 */
149     /**
150      * Creates all thumbnails and configure it as a Carousel.
151      * @private
152      * @function
153      * @name createThumbs
154      * @memberOf ch.Viewer
155      */
156 	var createThumbs = function(){
157 	
158 		var structure = $("<ul>").addClass("carousel");
159 		
160 		$.each(items, function(i, e){
161 			
162 			var thumb = $("<li>").bind("click", function(event){
163 				that.prevent(event);
164 				that.move(i + 1);
165 			});
166 			
167 			// Thumbnail
168 			if( $(e).children("link[rel=thumb]").length > 0 ) {
169 				$("<img>")
170 					.attr("src", $(e).children("link[rel=thumb]").attr("href"))
171 					.appendTo( thumb );
172 			
173 			// Google Map
174 			//} else if( ref.children("iframe").length > 0 ) {
175 				// Do something...
176 					
177 			// Video
178 			} else if( $(e).children("object").length > 0 || $(e).children("embed").length > 0 ) {
179 				$("<span>").html("Video").appendTo( thumb.addClass("ch-viewer-video") );
180 			};
181 			
182 			structure.append( thumb );
183 		});
184 		
185 		var self = {};
186 		
187 			self.children = structure.children();
188 			
189 			self.selected = 1;
190 		
191 			self.carousel = that.children[0] = $("<div>")
192 				.addClass("ch-viewer-triggers")
193 				.append( structure )
194 				.appendTo( $viewer )
195 				.carousel({ width: conf.width });
196 		
197 		return self;
198 	};
199 	
200     /**
201      * Moves the viewer's content.
202      * @private
203      * @function
204      * @name move
205      * @param {Number} item
206      * @returns {Chico-UI Object} that
207      * @memberOf ch.Viewer
208      */
209 	var move = function(item){
210 		// Validation
211 		if(item > itemsAmount || item < 1 || isNaN(item)) return that;
212 
213 		// Visual config
214 		$(thumbnails.children[thumbnails.selected - 1]).removeClass("ch-thumbnail-on"); // Disable thumbnail
215 		$(thumbnails.children[item - 1]).addClass("ch-thumbnail-on"); // Enable next thumbnail
216 		
217 		// Move Display carousel
218 		$display.moveTo(item);
219 		
220 		// Move thumbnails carousel if item selected is in other page
221 		var nextThumbsPage = Math.ceil( item / thumbnails.carousel.getSteps() );
222 		if(thumbnails.carousel.getPage() != nextThumbsPage) thumbnails.carousel.moveTo( nextThumbsPage );
223 		
224 		// Buttons behavior
225 		if(item > 1 && item < itemsAmount){
226 			arrows.prev.on();
227 			arrows.next.on();
228 		} else {
229 			if(item == 1){ arrows.prev.off(); arrows.next.on(); };
230 			if(item == itemsAmount){ arrows.next.off(); arrows.prev.on(); };
231 		};
232 		
233 		// Refresh selected thumb
234 		thumbnails.selected = item;
235 		
236 		// Callback
237 		that.callbacks("onMove");
238 	
239 		return that;
240 	};
241 	
242     /**
243      * Handles the visual behavior of arrows
244      * @private
245      * @name arrows
246      * @type {Object}
247      * @memberOf ch.Viewer
248      */
249  	var arrows = {};
250 	
251 	arrows.prev = {
252 		$element: $("<p>").addClass("ch-viewer-prev").bind("click", function(){ that.prev(); }),
253 		on: function(){ arrows.prev.$element.addClass("ch-viewer-prev-on") },
254 		off: function(){ arrows.prev.$element.removeClass("ch-viewer-prev-on") }
255 	};
256 	
257 	arrows.next = {
258 		$element: $("<p>").addClass("ch-viewer-next").bind("click", function(){ that.next(); }),
259 		on: function(){ arrows.next.$element.addClass("ch-viewer-next-on") },
260 		off: function(){ arrows.next.$element.removeClass("ch-viewer-next-on") }
261 	};
262 
263 /**
264  *  Protected Members
265  */ 
266 	
267 	that.prev = function(){
268 		that.move( thumbnails.selected - 1 );
269 		
270 		return that;
271 	};
272 	
273 	that.next = function(){
274 		that.move( thumbnails.selected + 1 );
275 		
276 		return that;
277 	};
278 
279 /**
280  *  Public Members
281  */	
282 
283     /**
284      * The component's instance unique identifier.
285      * @public
286      * @name uid
287      * @type {Number}
288      * @memberOf ch.Viewer
289      */
290 	that["public"].uid = that.uid;
291     /**
292      * The element reference.
293      * @public
294      * @name element
295      * @type {HTMLElement}
296      * @memberOf ch.Viewer
297      */
298 	that["public"].element = that.element;
299     /**
300      * The component's type.
301      * @public
302      * @name type
303      * @type {String}
304      * @memberOf ch.Viewer
305      */
306 	that["public"].type = that.type;
307     /**
308      * Children instances associated to this controller.
309      * @public
310      * @name children
311      * @type {Collection}
312      * @memberOf ch.Viewer
313      */
314  	that["public"].children = that.children;
315 	
316 	// Full behavior
317 	if(itemsAmount > 1) {
318         /**
319          * Selects a specific viewer's child.
320          * @public
321          * @function
322          * @name moveTo 
323          * @param {Number} item Recieve a index to select a children.
324          * @memberOf ch.Viewer
325          */
326         // TODO: This method should be called 'select'?
327 		that["public"].moveTo = function(item){ that.move(item); return that["public"]; };
328         /**
329          * Selects the next child available.
330          * @public
331          * @function
332          * @name next
333          * @memberOf ch.Viewer
334          */
335 		that["public"].next = function(){ that.next(); return that["public"]; };
336         /**
337          * Selects the previous child available.
338          * @public
339          * @function
340          * @name prev
341          * @memberOf ch.Viewer
342          */
343 		that["public"].prev = function(){ that.prev(); return that["public"]; };
344         /**
345          * Get the index of the selected child.
346          * @public
347          * @function
348          * @name getSelected
349          * @memberOf ch.Viewer
350          */
351 		that["public"].getSelected = function(){ return thumbnails.selected; }; // Is this necesary???
352 		// ...
353 
354 /**
355  *  Default event delegation
356  */
357 	
358 		// ...
359 		
360 		// Create thumbnails
361 		var thumbnails = createThumbs();
362 		
363 		// Create Viewer buttons
364 		$viewer.append( arrows.prev.$element ).append( arrows.next.$element );
365 		
366 		// Create movement method
367 		that.move = move;
368 		that.move(1); // Move to the first item without callback
369 		arrows.next.on();
370 	};
371 
372 	$viewer.find(".ch-mask").eq(0).height( $(itemsChildren[0]).height() );
373 
374 	ch.utils.avoidTextSelection(that.element);
375 
376 	return that;
377 };
