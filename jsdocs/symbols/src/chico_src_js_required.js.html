  2 /**
  3  * Required interface for Watcher.
  4  * @name Required
  5  * @class Required
  6  * @augments ch.Watcher
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  * @see ch.Number
 11  * @see ch.String
 12  * @see ch.Custom
 13  * @example
 14  * // Simple validation
 15  * $("input").required("This field is required");
 16  * @see ch.Watcher
 17  */
 18 
 19 ch.required = function(conf) {
 20 /**
 21  *  Constructor
 22  */
 23 	
 24 	conf = conf || {};
 25 	
 26     conf.messages = {};
 27 
 28     if ( conf.hasOwnProperty("msg") ) {     	
 29     	conf.messages.required = conf.msg;
 30     	conf.msg = null;
 31     	delete conf.msg;
 32     };
 33     
 34 	// Define the validation interface    
 35     conf.required = true;
 36     
 37     // Add validation types
 38 	conf.types = "required";
 39     // Define the conditions of this interface
 40     conf.conditions = [{
 41         name: "required"
 42     }];
 43 	
 44 	return ch.watcher.call(this, conf);
 45     
 46 };
