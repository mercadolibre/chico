  2 /**
  3  * Validate strings.
  4  * @name String
  5  * @class String
  6  * @augments ch.Watcher
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  * @see ch.Email
 11  * @see ch.Url
 12  * @see ch.MaxLength
 13  * @see ch.MinLength
 14  * @example
 15  * // Create a string validation
 16  * $("input").string("This field must be a string.");
 17  */
 18 
 19 ch.string = function(conf) {
 20 
 21 /**
 22  *  Constructor
 23  */
 24 	conf = conf || {};
 25 	
 26     conf.messages = conf.messages || {};
 27 
 28     if ( conf.hasOwnProperty("msg") ) { 
 29 		conf.messages.string = conf.msg;
 30     	conf.msg = null;
 31     	delete conf.msg;
 32     };
 33     
 34     // $.string("message"); support
 35     if ( !conf.text && !conf.email && !conf.url && !conf.maxLength && !conf.minLength ) {
 36         conf.text = true;
 37     };
 38 
 39 	// Add validation types
 40 	conf.types = "text,email,url,minLength,maxLength";
 41     
 42     // Define the conditions of this interface
 43     conf.conditions = [{
 44             name: "text", 
 45             patt: /^([a-zA-Z\s]+)$/ 
 46         },{
 47             name:"email",
 48             patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/ 
 49         },{
 50             name: "url",
 51             patt: /^((https?|ftp|file):\/\/|((www|ftp)\.)|(\/|.*\/)*)[a-z0-9-]+((\.|\/)[a-z0-9-]+)+([/?].*)?$/ 
 52         },{
 53             name: "minLength",
 54             expr: function(a,b) { return a.length >= b } 
 55         },{
 56             name: "maxLength",
 57             expr: function(a,b) { return a.length <= b } 
 58         }];
 59         // Conditions map TODO: uppercase, lowercase, varchar
 60 
 61 	return ch.watcher.call(this, conf);
 62     
 63 };
