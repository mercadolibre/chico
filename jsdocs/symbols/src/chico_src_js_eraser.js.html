  2 /**
  3  * Eraser is an utility to erase component's instances and free unused memory. A Numer will erase only that particular instance, a component's name will erase all components of that type, a "meltdown" will erase all component's instances from any kind.
  4  * @abstract
  5  * @name Eraser
  6  * @class Eraser
  7  * @memberOf ch
  8  * @param {String} [data] 
  9  */
 10  
 11 ch.eraser = function(data) {
 12     
 13     if(typeof data == "number"){
 14         
 15         // By UID
 16         for(var x in ch.instances){
 17             
 18             var component = ch.instances[x];
 19             
 20             for(var i = 0, j = component.length; i < j; i += 1){
 21                 if(component[i].uid == data){
 22                     // TODO: component.delete()
 23                     delete component[i];
 24                     component.splice(i, 1);
 25                     
 26                     return;
 27                 };
 28             };
 29         };
 30     
 31     } else {
 32         
 33         // All
 34         if(data === "meltdown"){
 35             // TODO: component.delete()
 36             /*for(var x in ch.instances){
 37                 var component = ch.instances[x];
 38                 for(var i = 0, j = component.length; i < j; i += 1){
 39                     component.delete();
 40                 };
 41             };*/
 42             
 43             delete ch.instances;
 44             ch.instances = {};
 45             
 46         // By component name    
 47         } else {
 48             
 49             for(var x in ch.instances){
 50             
 51                 if(x == data){
 52                     
 53                     var component = ch.instances[x];
 54                     
 55                     // TODO: component.delete()
 56                     /*for(var i = 0; i < component.length; i += 1){
 57                         component.delete()
 58                     };*/
 59                     
 60                     delete ch.instances[x];
 61                 };
 62             };
 63             
 64         };
 65         
 66     };
 67     
 68 };
