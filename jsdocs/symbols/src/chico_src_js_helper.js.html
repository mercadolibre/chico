  2 /**
  3  * Shows messages on the screen with a contextual floated UI-Component.
  4  * @name Helper
  5  * @class Helper
  6  * @augments ch.Floats
  7  * @memberOf ch
  8  * @param {Controller Object} o Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11 
 12 ch.helper = function(controller){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.Helper
 20      */
 21 	var that = this;
 22 
 23 	var conf = {};		
 24 		conf.cone = true;
 25 		conf.position = {};
 26 		conf.position.context = controller.reference;
 27 		conf.position.offset = "15 0";
 28 		conf.position.points = "lt rt";
 29 		conf.cache = false;
 30 	
 31 	that.conf = conf;
 32 
 33 /**
 34  *	Inheritance
 35  */
 36 
 37     that = ch.floats.call(that);
 38     that.parent = ch.clon(that);
 39 
 40 /**
 41  *  Private Members
 42  */
 43 
 44 
 45 
 46 /**
 47  *  Protected Members
 48  */ 
 49 	that.$trigger = that.$element;
 50 	
 51 	that.show = function(text) {
 52 
 53 		if ( !that.active ) {
 54 			// Load content and show!
 55 			conf.content = '<p><span class="ico error">Error: </span>' + text + '</p>';
 56 			that.parent.show();
 57 			
 58 		} else {
 59 			// Just Reload content!
 60 			that.$content.html('<p><span class="ico error">Error: </span>' + text + '</p>');
 61 			
 62 		};
 63 
 64 		return that;
 65 	};
 66 
 67 /**
 68  *  Public Members
 69  */
 70 
 71     /**
 72      * The component's instance unique identifier.
 73      * @public
 74      * @name uid
 75      * @type {Number}
 76      * @memberOf ch.Helper
 77      */ 
 78    	that["public"].uid = that.uid;
 79    	
 80     /**
 81      * The element reference.
 82      * @public
 83      * @name element
 84      * @type {HTMLElement}
 85      * @memberOf ch.Helper
 86      */
 87 	that["public"].element = that.element;
 88     /**
 89      * The component's type.
 90      * @public
 91      * @name type
 92      * @type {String}
 93      * @memberOf ch.Helper
 94      */
 95 	that["public"].type = that.type;
 96     /**
 97      * The component's content.
 98      * @public
 99      * @name content
100      * @type {String}
101      * @memberOf ch.Helper
102      */
103 	that["public"].content = that.content;
104     /**
105      * Shows component's content.
106      * @public
107      * @name show
108      * @function
109      * @returns {Chico-UI Object}
110      * @memberOf ch.Helper
111      */
112 	that["public"].show = function(text){
113 		that.show(text);
114 		
115 		return that["public"];
116 	};
117     /**
118      * Hides component's content.
119      * @public
120      * @name hide
121      * @function
122      * @returns {Chico-UI Object}
123      * @memberOf ch.Helper
124      */ 
125 	that["public"].hide = function(){
126 		that.hide();
127 		
128 		return that["public"];
129 	};
130     /**
131      * Positioning configuration.
132      * @public
133      * @name position
134      * @memberOf ch.Helper
135      * @example
136      * // Change helper's position.
137      * $('input').required("message").helper.position({ 
138      *    offset: "0 10",
139      *    points: "lt lb"
140      * });
141      */
142 	that["public"].position = that.position;
143 
144 
145 /**
146  *  Default event delegation
147  */
148 
149     $("body").bind(ch.events.LAYOUT.CHANGE, function(){ 
150         that.position("refresh");
151     });
152 
153 	 
154 	return that;
155 };
