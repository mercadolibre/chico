  2 /**
  3  * Zoom UI-Component for images.
  4  * @name Zoom
  5  * @class Zoom
  6  * @augments ch.Floats
  7  * @requires ch.Positioner
  8  * @requires ch.Preload
  9  * @memberOf ch
 10  * @param {Configuration Object} conf Object with configuration properties
 11  * @return {Chico-UI Object}
 12  */
 13 
 14 ch.zoom = function(conf) {
 15 
 16     /**
 17      * Reference to a internal component instance, saves all the information and configuration properties.
 18      * @private
 19      * @name that
 20      * @type {Object}
 21      * @memberOf ch.Zoom
 22      */
 23 	var that = this;
 24 	conf = ch.clon(conf);
 25 
 26 	// Link source as zoomed image
 27 	conf.content = $("<img>").attr("src", that.element.href);
 28 
 29 	conf.position = {};
 30 	conf.position.context = conf.context || that.$element;
 31 	conf.position.offset = conf.offset || "20 0";
 32 	conf.position.points = conf.points || "lt rt";
 33 	conf.position.hold = true;
 34 	
 35 	conf.width = conf.width || 300;
 36 	conf.height = conf.height || 300;
 37 	
 38 	conf.fx = false;
 39 
 40 	that.conf = conf;
 41 
 42 /**
 43  *	Inheritance
 44  */
 45 
 46     that = ch.floats.call(that);
 47     that.parent = ch.clon(that);
 48     
 49 /**
 50  *  Private Members
 51  */
 52 
 53     /**
 54      * Main configuration object.
 55      * @private
 56      * @name main
 57      * @type {Object}
 58      * @memberOf ch.Zoom
 59      */
 60 	var main = {};
 61 		main.img = that.$element.children();
 62 		main.w = main.img.width();
 63 		main.h = main.img.height();
 64 
 65     /**
 66      * The zoomed visual element.
 67      * @private
 68      * @name zoomed
 69      * @type {Object}
 70      * @memberOf ch.Zoom
 71      */
 72 	var zoomed = {};
 73 		zoomed.img = conf.content;
 74 	
 75 	// Magnifying glass
 76 	//var $lens = $("<div>").addClass("ch-lens ch-hide");
 77 	
 78     /**
 79      * The seeker is the visual element that follows mouse movement.
 80      * @private
 81      * @name seeker
 82      * @type {Object}
 83      * @memberOf ch.Zoom
 84      */
 85 	var seeker = {};
 86 		// TODO: Calc relativity like in that.size (en lugar de la division por 3)
 87 		seeker.w = conf.width / 3;
 88 		seeker.h = conf.height / 3;
 89 		seeker.shape = $("<div>")
 90 			.addClass("ch-seeker ch-hide")
 91 			.bind("mousemove", function(event){ move(event); })
 92 			.css({width: seeker.w, height: seeker.h});
 93     
 94     /**
 95      * Get the mouse position and moves the zoomed image.
 96      * @private
 97      * @function
 98      * @name move
 99      * @param {Mouse Event Object} event
100      * @memberOf ch.Zoom
101      */
102 	var move = function(event){
103 		var offset = main.img.offset();
104 		
105 		var x = event.pageX - offset.left;
106 		var y = event.pageY - offset.top;
107 		
108 		// Zoomed image
109 		zoomed.img.css({
110 			"left": -( ((zoomed.w * x) / main.w) - (conf.width / 2) ),
111 			"top": -( ((zoomed.h * y) / main.h) - (conf.height / 2) )
112 		});
113 		
114 		// Seeker shape
115 		seeker.shape.css({
116 			"left": x - (seeker.w / 2),
117 			"top": y - (seeker.h / 2)
118 		});
119 	};
120 	
121 /**
122  *  Protected Members
123  */
124 	
125 	that.$trigger = that.$element;
126 	
127 	that.show = function(event){
128 		that.prevent(event);
129 		
130 		// Floats show
131 		that.parent.show();
132 		
133 		// Magnifying glass
134 		//$lens.fadeIn();
135 		
136 		// Seeker
137 		seeker.shape.removeClass("ch-hide");
138 		
139 		return that;
140 	};
141 	
142 	that.hide = function(event){
143 		that.prevent(event);
144 		
145 		// Seeker
146 		seeker.shape.addClass("ch-hide");
147 		
148 		// Magnifying glass
149 		//$lens.fadeOut();
150 		
151 		// Floats hide
152 		that.parent.hide();
153 		
154 		return that;
155 	};
156 
157     /**
158      * Opens the big picture.
159      * @private
160      * @function
161      * @name enlarge
162      * @memberOf ch.Layer
163      */
164 	that.enlarge = function(event){
165 		that.prevent(event);
166 		
167 		// Open pop-up
168 	};
169 	
170     /**
171      * Getter and setter for size attributes.
172      * @private
173      * @function
174      * @name size
175      * @param {String} attr
176      * @param {String} [data]
177      * @memberOf ch.Layer
178      */
179 	that.size = function(attr, data) {
180 		if (!data) return conf[attr]; // Getter
181 		
182 		// Size of zoomed image
183 		zoomed.w = zoomed.img.width();
184 		zoomed.h = zoomed.img.height();
185 		zoomed.ref_x = (zoomed.w / main.w - 1);
186 		zoomed.ref_y = (zoomed.h / main.h - 1);
187 		
188 		var at = attr.substr(0,1);
189 		
190 		// Configuration
191 		that.conf[attr] = data;
192 		
193 		// Container
194 		that.$container[attr](data);
195 		
196 		// Seeker
197 		var rel = (main[at] * data) / zoomed[at];
198 		seeker[at] = rel;
199 		seeker.shape[attr](rel);
200 
201 		return that["public"];
202 	};
203 
204 /**
205  *  Public Members
206  */
207  
208     /**
209      * The component's instance unique identifier.
210      * @public
211      * @name uid
212      * @type {Number}
213      * @memberOf ch.Zoom
214      */
215    	that["public"].uid = that.uid;
216     /**
217      * The element reference.
218      * @public
219      * @name element
220      * @type {HTMLElement}
221      * @memberOf ch.Zoom
222      */
223 	that["public"].element = that.element;
224     /**
225      * The component's type.
226      * @public
227      * @name type
228      * @type {String}
229      * @memberOf ch.Zoom
230      */
231 	that["public"].type = that.type;
232     /**
233      * The component's content.
234      * @public
235      * @name content
236      * @type {String}
237      * @memberOf ch.Zoom
238      */
239 	that["public"].content = that.content;
240     /**
241      * Shows component's content.
242      * @public
243      * @name show
244      * @function
245      * @return {Chico-UI Object}
246      * @memberOf ch.Zoom
247      */
248 	that["public"].show = function(){
249 		that.show();
250 		
251 		return that["public"];
252 	};
253     /**
254      * Hides component's content.
255      * @public
256      * @name hide
257      * @function
258      * @return {Chico-UI Object}
259      * @memberOf ch.Zoom
260      */
261 	that["public"].hide = function(){
262 		that.hide();
263 		
264 		return that["public"];
265 	};
266 
267     /**
268      * Positioning configuration.
269      * @public
270      * @name position
271      * @memberOf ch.Zoom
272      * @example
273      * // Change position.
274      * $('input').zoom().position({ 
275      *    offset: "0 10",
276      *    points: "lt lb"
277      * });
278      */
279 	that["public"].position = that.position;
280 	
281     /**
282      * Sets the width size.
283      * @private
284      * @name width
285      * @param {Number} data Width value.
286      * @memberOf ch.Zoom
287      */
288 	that["public"].width = function(data){ that.size("width", data); };
289     /**
290      * Sets the height size.
291      * @private
292      * @name height
293      * @param {Number} data Height value.
294      * @memberOf ch.Zoom
295      */
296 	that["public"].height = function(data){ that.size("height", data) };
297 
298 	
299 /**
300  *  Default event delegation
301  */
302 	
303 	// TODO: El setTimeout soluciona problemas en el viewer
304 	setTimeout( function(){
305 		that.$element
306 			.addClass("ch-zoom-trigger")
307 			
308 			// Magnifying glass
309 			//.append( $lens )
310 			
311 			// Seeker
312 			.append( seeker.shape )
313 			
314 			// Size (same as image)
315 			.css({"width": main.w, "height": main.h})
316 			
317 			// Show
318 			.bind("mouseover", that.show)
319 			
320 			// Hide
321 			.bind("mouseleave", that.hide)
322 			
323 			// Move
324 			.bind("mousemove", function(event){ move(event); })
325 			
326 			// Enlarge
327 			.bind("click", function(event){ that.enlarge(event); });
328 	},50);	
329 	
330 	// Preload zoomed image
331 	if(ch.hasOwnProperty("preload")) ch.preload(that.element.href);
332 	
333 	return that;
334 };
