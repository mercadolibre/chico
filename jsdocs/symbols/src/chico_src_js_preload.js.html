  2 /**
  3  * Pre-load is an utility to preload images on browser's memory. An array of sources will iterate and preload each one, a single source will do the same thing.
  4  * @abstract
  5  * @name Preload
  6  * @class Preload
  7  * @memberOf ch
  8  * @param {Array} [arr] Collection of image sources
  9  * @param {String} [str] A single image source
 10  * @example
 11  * ch.preload(["img1.jpg","img2.jpg","img3.png"]);
 12  * @example
 13  * ch.preload("logo.jpg");
 14  */
 15 ch.preload = function(arr) {
 16 
 17     if (typeof arr === "string") {
 18         arr = (arr.indexOf(",") > 0) ? arr.split(",") : [arr] ;
 19     }
 20 
 21     for (var i=0;i<arr.length;i++) {
 22                 
 23         var o = document.createElement("object");
 24             o.data = arr[i]; // URL
 25             
 26         var h = document.getElementsByTagName("head")[0];
 27             h.appendChild(o);
 28             h.removeChild(o); 
 29     }       
 30 };
