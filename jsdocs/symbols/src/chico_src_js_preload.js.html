  2 /**
  3  * Pre-load is an utility to preload images on browser's memory. An array of sources will iterate and preload each one, a single source will do the same thing.
  4  * @name Preload
  5  * @class Preload
  6  * @memberOf ch
  7  * @param {Array} [arr] Collection of image sources
  8  * @param {String} [str] A single image source
  9  * @example
 10  * ch.preload(["img1.jpg","img2.jpg","img3.png"]);
 11  * @example
 12  * ch.preload("logo.jpg");
 13  */
 14 ch.preload = function(arr) {
 15 
 16     if (typeof arr === "string") {
 17         arr = (arr.indexOf(",") > 0) ? arr.split(",") : [arr] ;
 18     }
 19 
 20     for (var i=0;i<arr.length;i++) {
 21                 
 22         var o = document.createElement("object");
 23             o.data = arr[i]; // URL
 24             
 25         var h = document.getElementsByTagName("head")[0];
 26             h.appendChild(o);
 27             h.removeChild(o); 
 28     }       
 29 };
