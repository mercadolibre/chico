110     /**
111      * Creates a complete month and returns it in a table
112      * @private
113      * @function 
114      * @name name _createMonth
115      * @memberOf ch.Calendar
116      */
117 	var _createMonth = function(date){
118 
119 		var date = new Date(date);
120 
121 		var _tableMonth = _templateMonth.clone();
122 
123 		var _currentMonth = {};
124 			_currentMonth.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
125 			_currentMonth.date = _currentMonth.fullDate.getDate();
126 			_currentMonth.day = _currentMonth.fullDate.getDay();
127 			_currentMonth.month = _currentMonth.fullDate.getMonth();
128 			_currentMonth.year = _currentMonth.fullDate.getFullYear();
129 
130 
131 		var _currentDate = {};
132 			_currentDate.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
133 			_currentDate.date = _currentDate.fullDate.getDate();
134 			_currentDate.day = _currentDate.fullDate.getDay();
135 			_currentDate.month = _currentDate.fullDate.getMonth();
136 			_currentDate.year = _currentDate.fullDate.getFullYear();
137 
138 		var first_weekday = _currentMonth.day;
139 
140 		var _weeks, _classToday, _classSelected;
141 
142 		_weeks = "<tbody>";
143 		
144 		do {
145 			
146 			_weeks += "<tr class=\"week\">";
147 
148 			for (var i = 0; i < 7; i += 1) {
149 
150 				if (_currentDate.date == 1) {
151 					for (var i = 0; i < first_weekday; i += 1) {
152 						_weeks += "<td class=\"disable\"></td>";
153 					};
154 				};
155 				
156 				_classToday = (_currentDate.date == _today.getDate() && _currentDate.month == _today.getMonth() && _currentDate.year == _today.getFullYear()) ? " today" : "";
157 
158 				_classSelected = (_selected && _currentDate.date == _selected.getDate() && _currentDate.month == _selected.getMonth() && _currentDate.year == _selected.getFullYear()) ? " selected" : "";
159 				
160 				_weeks += "<td class=\"day" + _classToday +  _classSelected + "\">" + _currentDate.date + "</td>";
161 				
162 				_currentDate.fullDate.setDate(_currentDate.date+1);
163 				_currentDate.date = _currentDate.fullDate.getDate();
164 				_currentDate.day = _currentDate.fullDate.getDay();
165 				_currentDate.month = _currentDate.fullDate.getMonth();
166 				_currentDate.year = _currentDate.fullDate.getFullYear();
167 
168 				if ( _currentDate.month != _currentMonth.month ) { break; };
169 
170 			};
171 
172 			_weeks += "</tr>";
173 			
174 		} while (_currentDate.month == _currentMonth.month);
175 
176 		_weeks += "</tbody>";
177 
178 		_tableMonth
179 			.prepend("<caption>"+_monthsNames[_currentMonth.month] + " - " + _currentMonth.year+"</caption>")
180 			.append(_weeks);
181 
182 		$.each(_tableMonth.find(".day"), function(i, e){
183 			$(e).bind("click", function(){
184 				_select( _currentDate.year + "/" + _currentDate.month + "/" + this.innerHTML );
185 			});
186 		});
187 
188 		return _tableMonth;
189 	};

206     /**

227 			}
228 		});
229

231 	};

253      /**
254      * Parse Date object to YY/MM/DD format
255      * @private
256 	 * @function
257      * @name _parseDate
258      * @memberOf ch.Calendar
259      */
260 	var _parseDate = function(date){
261 
262 		if (typeof date == "undefined") { return; };
263 		
264 		return  date.getFullYear() + "/" + (date.getMonth() < 10 ? '0' : '') + (parseInt(date.getMonth(), 10) + 1) + "/" + (date.getDate() < 10 ? '0' : '') + date.getDate();
265 
266 	};
