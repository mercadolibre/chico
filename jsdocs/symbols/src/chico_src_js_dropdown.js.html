  2 /**
  3  * A navegable list of items, UI-Object.
  4  * @name Dropdown
  5  * @class Dropdown
  6  * @augments ch.Navs
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11 
 12 ch.dropdown = function(conf){
 13 
 14 
 15     /**
 16      * Reference to a internal component instance, saves all the information and configuration properties.
 17      * @private
 18      * @name that
 19      * @type {Object}
 20      * @memberOf ch.Dropdown
 21      */
 22 	var that = this;
 23 
 24 	conf = ch.clon(conf);
 25 	conf.skin = ( that.$element.hasClass("secondary") ) ? "secondary": "primary";
 26 
 27 	that.conf = conf;
 28 	
 29 /**
 30  *	Inheritance
 31  */
 32 
 33     that = ch.navs.call(that);
 34     that.parent = ch.clon(that);
 35 
 36 /**
 37  *  Protected Members
 38  */ 
 39 	that.$container = that.$element.addClass("ch-dropdown");
 40 	
 41 	that.$trigger = that.$container.children(":first");
 42 	
 43 	that.show = function(event){
 44 		that.prevent(event);
 45 		
 46 		// Toggle
 47 		if ( that.active ) {
 48 			return that.hide(event);
 49 		};
 50 		
 51         // Reset all dropdowns
 52 		$(ch.instances.dropdown).each(function(i, e){ e.hide(); });
 53 		
 54         // Show menu
 55 		that.$content.css('z-index', ch.utils.zIndex ++);
 56 		that.$container.css('z-index', ch.utils.zIndex ++); // Z-index of trigger over content		
 57 		that.parent.show(event);
 58 		that.position("refresh");
 59 		
 60 		// Secondary behavior
 61 		if(conf.skin == "secondary"){
 62 			that.$container.addClass("ch-dropdown-on"); // Container ON
 63 		};
 64 		
 65 		// Close events
 66 		ch.utils.document.one("click", function(event){ that.hide(event); });
 67 		ch.utils.document.bind(ch.events.KEY.ESC, function(event){ that.hide(event) });
 68 		
 69 		// Select first anchor child by default
 70 		var items = that.$content.find("a");
 71 			items.eq(0).focus();
 72 		
 73 		// More than one item
 74 		if(items.length == 0) return that;
 75 		
 76 		// Keyboard support
 77 		var itemSelected = 0;
 78 		
 79 		// Item selected by mouseover
 80 		// TODO: It's over keyboard selection and it is generating double selection.
 81 		items.each(function(i, e){
 82 			$(e).bind("mouseenter", function(){
 83 				itemSelected = i;
 84 				items.eq( itemSelected ).focus();
 85 			});
 86 		});
 87 		
 88 		var selectItem = function(arrow, event){
 89 			that.prevent(event);
 90 			
 91 			if(itemSelected == ((arrow == "bottom") ? items.length - 1 : 0)) return;
 92 			
 93 			items.eq( itemSelected ).blur();
 94 			
 95 			if(arrow == "bottom") itemSelected ++; else itemSelected --;
 96 			
 97 			items.eq( itemSelected ).focus();
 98 		};
 99 		
100 		// Arrows
101 		ch.utils.document.bind(ch.events.KEY.UP_ARROW, function(x, event){ selectItem("up", event); });
102 		ch.utils.document.bind(ch.events.KEY.DOWN_ARROW, function(x, event){ selectItem("bottom", event); });
103 		
104         return that;
105     };
106 	
107     that.hide = function(event){
108     	that.prevent(event);
109     	
110     	if (!that.active) return;
111     	
112     	// Secondary behavior
113 		if(conf.skin == "secondary"){
114 			that.$container.removeClass("ch-dropdown-on"); // Container OFF
115 		};
116 		
117         that.parent.hide(event);
118         
119         // Unbind events
120         ch.utils.document.unbind(ch.events.KEY.UP_ARROW);
121         ch.utils.document.unbind(ch.events.KEY.DOWN_ARROW);
122         ch.utils.document.unbind(ch.events.KEY.ESC);
123         
124         return that;
125 	};
126 	
127 /**
128  *  Public Members
129  */
130  
131     /**
132      * The component's instance unique identifier.
133      * @public
134      * @name uid
135      * @type {Number}
136      * @memberOf ch.Dropdown
137      */
138    	that["public"].uid = that.uid;
139     
140     /**
141      * The element reference.
142      * @public
143      * @name element
144      * @type {HTMLElement}
145      * @memberOf ch.Dropdown
146      */
147 	that["public"].element = that.element;
148     
149     /**
150      * The component's type.
151      * @public
152      * @name type
153      * @type {String}
154      * @memberOf ch.Dropdown
155      */	
156 	that["public"].type = that.type;
157 	
158     /**
159      * Shows component's content.
160      * @public
161      * @name show
162      * @return {Chico-UI Object}
163      * @memberOf ch.Dropdown
164      */
165 	that["public"].show = function(){
166 		that.show();
167 		
168 		return that["public"];
169 	};
170 
171     /**
172      * Hides component's content.
173      * @public
174      * @name hide
175      * @return {Chico-UI Object}
176      * @memberOf ch.Dropdown
177      */ 
178 	that["public"].hide = function(){
179 		that.hide();
180 		
181 		return that["public"];
182 	};
183     /**
184      * Positioning configuration.
185      * @public
186      * @name position
187      * @memberOf ch.Dropdown
188      */
189 	that["public"].position = that.position;	
190 
191 
192 /**
193  *  Default event delegation
194  */		    
195     // Trigger
196 	that.$trigger
197 		.bind("click", function(event){ that.show(event) })
198 		.addClass("ch-dropdown-trigger-" + conf.skin)
199 		.append("<span class=\"ch-bottom\"> »</span>");
200 
201 	// Content
202 	that.$content = that.$trigger.next()
203 		.addClass("ch-dropdown-content-" + conf.skin)
204 		.addClass("ch-hide")
205 		.bind("click", function(event){
206 			event.stopPropagation(); // Prevent click on content (except links)
207 		})
208 		.detach(); // Save on memory;
209 	
210 	// Close dropdown after click an option (link)
211 	that.$content.find("a").one("click", function(){ that.hide(); });
212 
213 	// Put content out of container
214 	that.$container.after( that.$content );
215 		
216 	// Position
217 	that.conf.position = {};
218 	that.conf.position.element = that.$content;
219 	that.conf.position.context = that.$trigger;
220 	that.conf.position.points = "lt lb";
221 	that.conf.position.offset = "0 -1";
222 	
223 	return that;
224 
225 };
