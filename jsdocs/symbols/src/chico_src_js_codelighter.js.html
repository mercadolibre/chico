  2 /** 
  3  * A simple utility to highlight code snippets in the HTML.
  4  * @abstract
  5  * @name Codelighter
  6  * @class Codelighter 
  7  * @return {Object}
  8  * @memberOf ch 
  9  * @example
 10  * ch.codelighter();
 11  * $(".xml").codeXML();
 12  */
 13 
 14 ch.codelighter = function() {
 15 /**
 16  *  Constructor
 17  */
 18 	var that = this;
 19 		
 20 /**
 21  *  Inheritance
 22  */
 23 
 24     that = ch.controllers.call(that);
 25     that.parent = ch.clon(that);
 26 	
 27 	
 28 /**
 29  *  Private Members
 30  */
 31 
 32 /**
 33  *  Protected Members
 34  */ 
 35 
 36 /**
 37  *  Public Members
 38  */
 39 	
 40 	that["public"].uid = that.uid;
 41 	that["public"].element = that.element;
 42 	that["public"].type = that.type;	
 43 	that["public"].children = that.children;	
 44 
 45 
 46 /**
 47  *  Default event delegation
 48  */	
 49 
 50 	$("pre[name=code]").each(function(i, e){
 51 		
 52 		var codesnippet = {};
 53 			codesnippet.uid = ch.utils.index += 1;
 54 			codesnippet.type = "codesnippet";
 55 			codesnippet.element = e;
 56 			codesnippet.snippet = e.innerHTML;			
 57 
 58 		that.children.push( ch["code" + e.className.toUpperCase()].call(codesnippet) );
 59 
 60 	});
 61 
 62 	ch.instances.codelighter = that.children; // Create codeligther instance
 63 	
 64 	return that;
 65 };

 69 /**
 70  *	@Codesnippet
 71  */
 72  
 73 ch.codesnippet = function(conf){
 74 /**
 75  *  Constructor
 76  */
 77 	
 78 	var that = this;	
 79 	
 80 	conf = ch.clon(conf);
 81 	that.conf = conf;
 82 	
 83 		
 84 /**
 85  *  Inheritance
 86  */
 87 
 88     that = ch.object.call(that);
 89     that.parent = ch.clon(that);
 90 	
 91 /**
 92  *  Private Members
 93  */
 94 
 95 	var paint = function() {
 96 		for (var x in conf.brush){
 97 			if (conf.brush[ x ].test(that.paintedSnippet)) {
 98 				that.paintedSnippet = that.paintedSnippet.replace(conf.brush[ x ], x);
 99 			};
100 		};
101 
102 		return that.paintedSnippet;
103 	};
104  
105 /**
106  *  Protected Members
107  */ 
108 
109 	that.paintedSnippet = that.snippet;	
110 			
111 /**
112  *  Public Members
113  */	
114 
115 	that["public"] = {};
116 	that["public"].uid = that.uid;
117 	that["public"].type = conf.type;
118 	that["public"].element = that.element;
119 	that["public"].snippet = that.snippet;
120 	that["public"].paintedSnippet = that.paintedSnippet;	
121 
122 /**
123  *  Default event delegation
124  */		 	
125 	
126 	that.element.innerHTML = paint();
127 	
128 	return that;
129 };

137 ch.codeXML = function(conf) {
138     
139 /** 
140  *  Constructor: Redefine or preset component's settings
141  */
142 
143 	conf = conf || {};
144 	
145 	conf.type = "codeXML";
146 
147 	conf.brush = {
148 		"<": /</g, // Menor
149 		">": />/g , // Mayor
150 		"<span class='ch-comment'>$&</span>": /(\<|<)!--\s*.*?\s*--(\>|>)/g, // comments		
151 		"<span class='ch-attrName'>$&</span>": /(id|name|class|title|alt|value|type|style|method|href|action|lang|dir|src|tabindex|usemap|data|rel|charset|encoding|size|selected|checked|placeholder|target|required|disabled|max|min|maxlength|accesskey)=".*"/g, // Attributes name
152 		"<span class='ch-attrValue'>$&</span>": /".+?"/g, // Attributes
153 		"<span class='ch-tag'>$&</span>": /(<([a-z]|\/).*?>)/g, // Tag
154 		"    ": /\t/g // Tab
155 	};
156     
157     this.snippet = this.snippet || this.element.innerHTML;
158     
159     return ch.codesnippet.call(this, conf);
160     
161 };
162 
163 ch.factory({ component: 'codeXML' });

172 ch.codeJS = function(conf) {
173     
174 /** 
175  *  Constructor: Redefine or preset component's settings
176  */
177 	
178 	conf = conf || {};
179 	
180 	conf.type = "codeJS";
181 	
182 	conf.brush = {
183 		"$1 $2 $3": /(<)([a-z]|\/|.*?)(>)/g,
184 		"<span class='ch-operator'>$&</span>": /(\+|\-|=|\*|&|\||\%|\!|\?)/g,
185 		">": />amp;/g,
186 		"<span class='ch-atom'>$&</span>": /(false|null|true|undefined)/g,		
187 		"$1<span class='ch-keywords'>$2</span>$3": /(^|\s|\(|\{)(return|new|delete|throw|else|case|break|case|catch|const|continue|default|delete|do|else|finally|for|function|if|in|instanceof|new|switch|throw|try|typeof|var|void|while|with)(\s*)/g,
188 		"<span class='ch-attrValue'>$&</span>": /(".+?")|[0-9]/g, // Attributes & numbers
189 		"    ": /\t/g, // Tab
190 		"<span class='ch-comment'>$&</span>": /(\/\*)\s*.*\s*(\*\/)/g, // Comments
191 		"<span class='ch-comment'>$&</span>": /(\/\/)\s*.*\s*\n*/g // Comments
192 		
193 	};
194 	    
195     this.snippet = this.snippet || this.element.innerHTML;
196     
197     return ch.codesnippet.call(this, conf);
198     
199 };

201 ch.factory({ component: 'codeJS' });
