  2 /**
  3  * Expando is a UI-Component.
  4  * @name Expando
  5  * @class Expando
  6  * @augments ch.Navs
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11  
 12 ch.expando = function(conf){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.Expando
 20      */
 21     var that = this;
 22 		
 23 	conf = ch.clon(conf);
 24 	that.conf = conf;
 25 	
 26 /**
 27  *	Inheritance
 28  */
 29 
 30 	that = ch.navs.call(that);
 31 	that.parent = ch.clon(that);
 32 
 33 /**
 34  *  Protected Members
 35  */ 
 36 
 37 	that.$trigger = that.$element.children().eq(0).wrapInner("<span>").children();
 38 
 39 	that.$content = that.$element.children().eq(1);
 40 
 41 /**
 42  *  Public Members
 43  */
 44  
 45     /**
 46      * The component's instance unique identifier.
 47      * @public
 48      * @name uid
 49      * @type {Number}
 50      * @memberOf ch.Expando
 51      */
 52    	that["public"].uid = that.uid;
 53     
 54     /**
 55      * The element reference.
 56      * @public
 57      * @name element
 58      * @type {HTMLElement}
 59      * @memberOf ch.Expando
 60      */
 61  	that["public"].element = that.element;
 62  	
 63     /**
 64      * The component's type.
 65      * @public
 66      * @name type
 67      * @type {String}
 68      * @memberOf ch.Expando
 69      */
 70 	that["public"].type = that.type;
 71 	
 72     /**
 73      * Shows component's content.
 74      * @public
 75      * @name show
 76      * @returns {Chico-UI Object}
 77      * @memberOf ch.Expando
 78      */
 79 	that["public"].show = function(){
 80 		that.show();
 81 		
 82 		return that["public"];
 83 	};
 84 
 85     /**
 86      * Hides component's content.
 87      * @public
 88      * @name hide
 89      * @returns {Chico-UI Object}
 90      * @memberOf ch.Expando
 91      */	
 92 	that["public"].hide = function(){
 93 		that.hide();
 94 		
 95 		return that["public"];
 96 	};
 97 	
 98 
 99 /**
100  *  Default event delegation
101  */		
102     
103 	that.configBehavior();
104 	ch.utils.avoidTextSelection(that.$trigger);
105 
106 	return that;
107 
108 };
