  2 /**
  3  * Abstract representation of navs components.
  4  * @abstract
  5  * @name Navs
  6  * @class Navs
  7  * @augments ch.Object
  8  * @memberOf ch
  9  * @param {Configuration Object} conf Object with configuration properties
 10  * @return {Chico-UI Object}
 11  * @see ch.Dropdown
 12  * @see ch.Expando
 13  */
 14  
 15 ch.navs = function(){
 16 	
 17     /**
 18      * Reference to a internal component instance, saves all the information and configuration properties.
 19      * @private
 20      * @name that
 21      * @type {Object}
 22      * @memberOf ch.Navs
 23      */ 
 24 	var that = this;
 25 	var conf = that.conf;
 26 /**
 27  *  Inheritance
 28  */
 29 
 30     that = ch.object.call(that);
 31     that.parent = ch.clon(that);
 32 			
 33 /**
 34  *  Public Members
 35  */ 	
 36 	that.active = false;
 37 
 38     /**
 39      * Shows component's content.
 40      * @public
 41      * @name show
 42      * @return {Chico-UI Object}
 43      * @memberOf ch.Navs
 44      */
 45 	that.show = function(event){
 46 		that.prevent(event);
 47 		
 48 		if ( that.active ) return;
 49 		
 50 		that.active = true;
 51 
 52 		that.$trigger.addClass("ch-" + that.type + "-on");
 53 		that.$content.removeClass("ch-hide");
 54 		that.callbacks("onShow");
 55 		
 56 		return that;
 57 	};
 58     /**
 59      * Hides component's content.
 60      * @public
 61      * @name hide
 62      * @return {Chico-UI Object}
 63      * @memberOf ch.Navs
 64      */
 65 	that.hide = function(event){
 66 		that.prevent(event);
 67 		
 68 		if (!that.active) return;
 69 		
 70 		that.active = false;
 71 		
 72 		that.$trigger.removeClass("ch-" + that.type + "-on");
 73 		that.$content.addClass("ch-hide");
 74 		that.callbacks("onHide");
 75 		
 76 		return that;
 77 	};		
 78 	
 79 	return that;
 80 }
