  2 /**
  3  * Zoom UI-Component for images.
  4  * @name Zoom
  5  * @class Zoom
  6  * @augments ch.Floats
  7  * @requires ch.Positioner
  8  * @memberOf ch
  9  * @param {Configuration Object} conf Object with configuration properties
 10  * @returns {Chico-UI Object}
 11  */
 12 
 13 ch.zoom = function(conf) {
 14 
 15     /**
 16      * Reference to an internal component instance, saves all the information and configuration properties.
 17      * @private
 18      * @name that
 19      * @type {Object}
 20      * @memberOf ch.Zoom
 21      */
 22 	var that = this;
 23 
 24 	conf = ch.clon(conf);
 25 	conf.width = conf.width || 300;
 26 	conf.height = conf.height || 300;
 27 	conf.fx = false;
 28 	conf.position = {};
 29 	conf.position.context = conf.context || that.$element;
 30 	conf.position.offset = conf.offset || "20 0";
 31 	conf.position.points = conf.points || "lt rt";
 32 	conf.position.hold = true;
 33 
 34 	that.conf = conf;
 35 
 36 /**
 37  *	Inheritance
 38  */
 39 
 40     that = ch.floats.call(that);
 41     that.parent = ch.clon(that);
 42     
 43 /**
 44  *  Private Members
 45  */
 46 
 47     /**
 48      * Original image.
 49      * @private
 50      * @name original
 51      * @type {Object}
 52      * @memberOf ch.Zoom
 53      */
 54 	var original = {};
 55 		original.img = that.$element.children();
 56 		original["width"] = original.img.prop("width");
 57 		original["height"] = original.img.prop("height");
 58 
 59     /**
 60      * Zoomed visual element.
 61      * @private
 62      * @name zoomed
 63      * @typeÂ {Object}
 64      * @memberOf ch.Zoom
 65      */
 66 	var zoomed = {};
 67 		zoomed.img = conf.content = $("<img>");
 68 	
 69 	// Magnifying glass (enlarge)
 70 	//var $lens = $("<div>").addClass("ch-lens ch-hide");
 71 	
 72     /**
 73      * Seeker is the visual element that follows mouse movement for referencing to zoomable area into original image.
 74      * @private
 75      * @name seeker
 76      * @typeÂ {Object}
 77      * @memberOf ch.Zoom
 78      */
 79 	var seeker = {};
 80 		seeker.shape = $("<div>")
 81 			.addClass("ch-seeker ch-hide")
 82 			.bind("mousemove", function(event){ move(event); })
 83 			// TODO: Calc relativity like in that.size (en lugar de la division por 3)
 84 			.css({ width: conf.width / 3, height: conf.height / 3 });
 85     
 86     /**
 87      * Get the mouse position and moves the zoomed image.
 88      * @private
 89      * @function
 90      * @name move
 91      * @param {Mouse Event Object} event
 92      * @memberOf ch.Zoom
 93      */
 94 	var move = function(event){
 95 		var offset = original.img.offset();
 96 		
 97 		var x = event.pageX - offset.left;
 98 		var y = event.pageY - offset.top;
 99 		
100 		// Zoomed image
101 		zoomed.img.css({
102 			"left": -( ((zoomed["width"] * x) / original["width"]) - (conf.width / 2) ),
103 			"top": -( ((zoomed["height"] * y) / original["height"]) - (conf.height / 2) )
104 		});
105 		
106 		// Seeker shape
107 		seeker.shape.css({
108 			"left": x - seeker["width"],
109 			"top": y - seeker["height"]
110 		});
111 	};
112 	
113 /**
114  *  Protected Members
115  */
116 	
117 	that.$trigger = that.$element;
118 	
119 	that.show = function(event){
120 		that.prevent(event);
121 
122 		zoomed.img.prop("src", that.element.href);
123 
124 		// Floats show
125 		that.parent.show();
126 
127 		// Magnifying glass
128 		//$lens.fadeIn();
129 
130 		// Seeker
131 		seeker.shape.removeClass("ch-hide");
132 		
133 		return that;
134 	};
135 	
136 	that.hide = function(event){
137 		that.prevent(event);
138 		
139 		// Seeker
140 		seeker.shape.addClass("ch-hide");
141 		
142 		// Magnifying glass
143 		//$lens.fadeOut();
144 		
145 		// Floats hide
146 		that.parent.hide();
147 		
148 		return that;
149 	};
150 
151     /**
152      * Opens the big picture.
153      * @private
154      * @function
155      * @name enlarge
156      * @memberOf ch.Layer
157      */
158 	that.enlarge = function(event){
159 		that.prevent(event);
160 		
161 		// Open pop-up
162 	};
163 	
164     /**
165      * Getter and setter for size attributes.
166      * @private
167      * @function
168      * @name size
169      * @param {String} attr
170      * @param {String} [data]
171      * @memberOf ch.Layer
172      */
173 	that.size = function(attr, data) {
174 		if (!data) return conf[attr]; // Getter
175 
176 		// Configuration
177 		that.conf[attr] = data;
178 		
179 		// Container
180 		that.$container[attr](data);
181 		
182 		// Seeker
183 		var size = (original[attr] * data) / zoomed[attr]; // Shape size relative to zoomed image and zoomed area
184 		seeker.shape[attr](size); // Sets shape size
185 		seeker[attr] = size / 2; // Shape half size: for position it
186 
187 		return that["public"];
188 	};
189 
190 /**
191  *  Public Members
192  */
193  
194     /**
195      * The component's instance unique identifier.
196      * @public
197      * @name uid
198      * @type {Number}
199      * @memberOf ch.Zoom
200      */
201    	that["public"].uid = that.uid;
202     /**
203      * The element reference.
204      * @public
205      * @name element
206      * @type {HTMLElement}
207      * @memberOf ch.Zoom
208      */
209 	that["public"].element = that.element;
210     /**
211      * The component's type.
212      * @public
213      * @name type
214      * @type {String}
215      * @memberOf ch.Zoom
216      */
217 	that["public"].type = that.type;
218     /**
219      * The component's content.
220      * @public
221      * @name content
222      * @type {String}
223      * @memberOf ch.Zoom
224      */
225 	that["public"].content = that.content;
226     /**
227      * Shows component's content.
228      * @public
229      * @name show
230      * @function
231      * @returns {Chico-UI Object}
232      * @memberOf ch.Zoom
233      */
234 	that["public"].show = function(){
235 		that.show();
236 		
237 		return that["public"];
238 	};
239     /**
240      * Hides component's content.
241      * @public
242      * @name hide
243      * @function
244      * @returns {Chico-UI Object}
245      * @memberOf ch.Zoom
246      */
247 	that["public"].hide = function(){
248 		that.hide();
249 		
250 		return that["public"];
251 	};
252 
253     /**
254      * Positioning configuration.
255      * @public
256      * @name position
257      * @memberOf ch.Zoom
258      * @example
259      * // Change position.
260      * $('input').zoom().position({ 
261      *    offset: "0 10",
262      *    points: "lt lb"
263      * });
264      */
265 	that["public"].position = that.position;
266 	
267     /**
268      * Gets and sets the width size.
269      * @private
270      * @name width
271      * @param {Number} data Width value.
272      * @memberOf ch.Zoom
273      * @example
274      * // Gets width of zoomed visual element.
275      * foo.width();
276      * @example
277      * // Sets width of zoomed visual element and update the seeker size to keep these relation.
278      * foo.width(500);
279      */
280 	that["public"].width = function(data){ that.size("width", data); };
281     /**
282      * Gets and sets the height size.
283      * @private
284      * @name height
285      * @param {Number} data Height value.
286      * @memberOf ch.Zoom
287      * @example
288      * // Gets height of zoomed visual element.
289      * foo.height();
290      * @example
291      * // Sets height of zoomed visual element and update the seeker size to keep these relation.
292      * foo.height(500);
293      */
294 	that["public"].height = function(data){ that.size("height", data) };
295 
296 	
297 /**
298  *  Default event delegation
299  */
300 	
301 	// TODO: El setTimeout soluciona problemas en el viewer
302 	setTimeout( function(){
303 		that.$element
304 			.addClass("ch-zoom-trigger")
305 			
306 			// Magnifying glass
307 			//.append( $lens )
308 			
309 			// Seeker
310 			.append( seeker.shape )
311 			
312 			// Size (same as image)
313 			.css({"width": original["width"], "height": original["height"]})
314 			
315 			// Show
316 			.bind("mouseenter", that.show)
317 			
318 			// Hide
319 			.bind("mouseleave", that.hide)
320 			
321 			// Move
322 			.bind("mousemove", function(event){ move(event); })
323 			
324 			// Enlarge
325 			.bind("click", function(event){ that.enlarge(event); });
326 	},50);	
327 
328 	// Preload zoomed image
329 	ch.utils.window.one("load", function(){
330 		zoomed.img.prop("src", that.element.href).one("load", function(){
331 			zoomed["width"] = zoomed.img.prop("width");
332 			zoomed["height"] = zoomed.img.prop("height");
333 		});
334 	});
335 
336 
337 	return that;
338 };
