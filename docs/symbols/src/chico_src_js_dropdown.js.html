  2 /**
  3  * A navegable list of items, UI-Object.
  4  * @name Dropdown
  5  * @class Dropdown
  6  * @augments ch.Navs
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11 
 12 ch.dropdown = function(conf){
 13 
 14 
 15     /**
 16      * Reference to a internal component instance, saves all the information and configuration properties.
 17      * @private
 18      * @name that
 19      * @type {Object}
 20      * @memberOf ch.Dropdown
 21      */
 22 	var that = this;
 23 
 24 	conf = ch.clon(conf);
 25 	that.conf = conf;
 26 	
 27 /**
 28  *	Inheritance
 29  */
 30 
 31     that = ch.navs.call(that);
 32     that.parent = ch.clon(that);
 33 
 34 /**
 35  *  Private Members
 36  */
 37     /**
 38      * Adds keyboard events.
 39      * @private
 40      * @name shortcuts
 41      * @function
 42      * @memberOf ch.TabNavigator
 43      */
 44 	var shortcuts = function(items){
 45 		
 46 		// Keyboard support
 47 		var selected = 0;
 48 		
 49 		// Item selected by mouseover
 50 		// TODO: It's over keyboard selection and it is generating double selection.
 51 		$.each(items, function(i, e){
 52 			$(e).bind("mouseenter", function(){
 53 				selected = i;
 54 				items.eq( selected ).focus();
 55 			});
 56 		});
 57 		
 58 		var selectItem = function(arrow, event){
 59 			that.prevent(event);
 60 			
 61 			if(selected == ((arrow == "bottom") ? items.length - 1 : 0)) return;
 62 			
 63 			items.eq( selected ).blur();
 64 			
 65 			if(arrow == "bottom") selected += 1; else selected -= 1;
 66 			
 67 			items.eq( selected ).focus();
 68 		};
 69 		
 70 		// Arrows
 71 		ch.utils.document.bind(ch.events.KEY.UP_ARROW, function(x, event){ selectItem("up", event); });
 72 		ch.utils.document.bind(ch.events.KEY.DOWN_ARROW, function(x, event){ selectItem("bottom", event); });
 73 	};
 74 
 75 
 76 /**
 77  *  Protected Members
 78  */
 79     /**
 80      * The component's trigger.
 81      * @private
 82      * @name $trigger
 83      * @type {jQuery Object}
 84      * @memberOf ch.Dropdown
 85      */
 86 	that.$trigger = that.$element.children().eq(0);
 87     /**
 88      * The component's content.
 89      * @private
 90      * @name $content
 91      * @type {jQuery Object}
 92      * @memberOf ch.Dropdown
 93      */
 94 	that.$content = that.$trigger.next().detach(); // Save on memory;
 95 
 96 	that.show = function(event){
 97 		that.prevent(event);
 98 
 99 		that.$content.css('z-index', ch.utils.zIndex ++);
100 		
101 		if (that.$element.hasClass("secondary")) { // Z-index of trigger over conten 
102 			that.$trigger.css('z-index', ch.utils.zIndex ++);
103 		}; 
104 
105 		
106 		that.$element
107 			.addClass("ch-dropdown-on")
108 			.css('z-index', ch.utils.zIndex ++);
109 
110 		that.parent.show(event);
111 		that.position("refresh");
112 
113 		// Reset all dropdowns
114 		$.each(ch.instances.dropdown, function(i, e){ 
115 			if (e.uid !== that.uid) e.hide();
116 		});
117 
118 		// Close events
119 		ch.utils.document.one("click " + ch.events.KEY.ESC, function(event){ that.hide(event); });
120 		// Close dropdown after click an option (link)
121 		that.$content.find("a").one("click", function(){ that.hide(); });
122 
123 		// Keyboard support
124 		var items = that.$content.find("a");
125 			items.eq(0).focus(); // Select first anchor child by default
126 
127 		if (items.length > 1){ shortcuts(items); };
128 
129 		return that;
130 	};
131 
132 	that.hide = function(event){
133 		that.prevent(event);
134 
135         that.parent.hide(event);
136         	that.$element.removeClass("ch-dropdown-on");
137 
138         // Unbind events
139         ch.utils.document.unbind(ch.events.KEY.ESC + " " + ch.events.KEY.UP_ARROW + " " + ch.events.KEY.DOWN_ARROW);
140 
141 		return that;
142 	};
143 	
144 /**
145  *  Public Members
146  */
147  
148     /**
149      * The component's instance unique identifier.
150      * @public
151      * @name uid
152      * @type {Number}
153      * @memberOf ch.Dropdown
154      */
155    	that["public"].uid = that.uid;
156     
157     /**
158      * The element reference.
159      * @public
160      * @name element
161      * @type {HTMLElement}
162      * @memberOf ch.Dropdown
163      */
164 	that["public"].element = that.element;
165     
166     /**
167      * The component's type.
168      * @public
169      * @name type
170      * @type {String}
171      * @memberOf ch.Dropdown
172      */	
173 	that["public"].type = that.type;
174 	
175     /**
176      * Shows component's content.
177      * @public
178      * @name show
179      * @returns {Chico-UI Object}
180      * @memberOf ch.Dropdown
181      */
182 	that["public"].show = function(){
183 		that.show();
184 		
185 		return that["public"];
186 	};
187 
188     /**
189      * Hides component's content.
190      * @public
191      * @name hide
192      * @returns {Chico-UI Object}
193      * @memberOf ch.Dropdown
194      */ 
195 	that["public"].hide = function(){
196 		that.hide();
197 		
198 		return that["public"];
199 	};
200     /**
201      * Positioning configuration.
202      * @public
203      * @name position
204      * @memberOf ch.Dropdown
205      */
206 	that["public"].position = that.position;	
207 
208 
209 /**
210  *  Default event delegation
211  */		    
212 
213 	that.configBehavior();
214 	
215 	that.$element.after( that.$content ); // Put content out of element
216 	ch.utils.avoidTextSelection(that.$trigger);
217 	
218 	if (that.$element.hasClass("secondary")) that.$content.addClass("secondary");
219 	
220 	// Prevent click on content (except links)
221 	that.$content.bind("click", function(event){ event.stopPropagation(); });
222 	
223 	// Position
224 	that.conf.position = {};
225 	that.conf.position.element = that.$content;
226 	that.conf.position.context = that.$trigger;
227 	that.conf.position.points = "lt lb";
228 	that.conf.position.offset = "0 -1";
229 
230 	return that;
231 
232 };
