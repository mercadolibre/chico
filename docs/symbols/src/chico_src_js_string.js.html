  2 /**
  3  * Validate strings.
  4  * @name String
  5  * @class String
  6  * @augments ch.Watcher
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  * @see ch.Watcher
 11  * @example
 12  * // Create a string validation
 13  * $("input").string("This field must be a string.");
 14  */
 15 
 16 ch.string = function(conf) {
 17 
 18 /**
 19  *  Constructor
 20  */
 21 	conf = conf || {};
 22 	
 23     conf.messages = conf.messages || {};
 24 
 25     if ( ch.utils.hasOwn(conf, "msg") ) { 
 26 		conf.messages.string = conf.msg;
 27     	conf.msg = null;
 28     	delete conf.msg;
 29     };
 30     
 31     // $.string("message"); support
 32     if ( !conf.text && !conf.email && !conf.url && !conf.maxLength && !conf.minLength ) {
 33         conf.text = true;
 34     };
 35 
 36 	// Add validation types
 37 	conf.types = "text,email,url,minLength,maxLength";
 38     
 39     // Define the conditions of this interface
 40     conf.conditions = [{
 41             name: "text", 
 42             patt: /^([a-zA-Z\s]+)$/
 43         },{
 44             name:"email",
 45             patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/ 
 46         },{
 47             name: "url",
 48             patt: /^((https?|ftp|file):\/\/|((www|ftp)\.)|(\/|.*\/)*)[a-z0-9-]+((\.|\/)[a-z0-9-]+)+([/?].*)?$/ 
 49         },{
 50             name: "minLength",
 51             expr: function(a,b) { return a.length >= b } 
 52         },{
 53             name: "maxLength",
 54             expr: function(a,b) { return a.length <= b } 
 55         }];
 56         // Conditions map TODO: uppercase, lowercase, varchar
 57 
 58     return ch.watcher.call(this, conf);        
 59 
 60 };
