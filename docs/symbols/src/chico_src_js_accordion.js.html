  2 /**
  3  * Accordion is a UI-Component.
  4  * @name Accordion
  5  * @class Accordion
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11  
 12 ch.accordion = function(conf){
 13 
 14 // Private members
 15 
 16     /**
 17      * Reference to a internal component instance, saves all the information and configuration properties.
 18      * @private
 19      * @name that
 20      * @type {Object}
 21      * @memberOf ch.Accordion
 22      */	
 23 	var that = this;
 24 	that.$element.addClass('ch-accordion');
 25 	conf = ch.clon(conf);
 26 	that.conf = conf;
 27 	
 28     // Inheritance
 29     that = ch.controllers.call(that);
 30     that.parent = ch.clon(that);
 31 
 32     /**
 33      * The component's instance unique identifier.
 34      * @public
 35      * @name uid
 36      * @type {Number}
 37      * @memberOf ch.Accordion
 38      */ 	
 39 	that["public"].uid = that.uid;
 40 	
 41     /**
 42      * The element reference.
 43      * @public
 44      * @name element
 45      * @type {HTMLElement}
 46      * @memberOf ch.Accordion
 47      */
 48 	that["public"].element = that.element;
 49 	
 50     /**
 51      * The component's type.
 52      * @public
 53      * @name type
 54      * @type {String}
 55      * @memberOf ch.Accordion
 56      */
 57 	that["public"].type = that.type;
 58 	
 59     /**
 60      * Select a specific children.
 61      * @public
 62      * @name select
 63      * @function
 64      * @memberOf ch.Accordion
 65      */
 66 	that["public"].select = function(bellows){
 67 		
 68 		if(typeof bellows == "string") {
 69 			var sliced = bellows.split("#");
 70 		
 71 			that.children[ sliced[0] ].select( sliced[1] );
 72 		} else {
 73 			that.children[ bellows ].show();
 74 		};
 75 		
 76 		that.callbacks("onSelect");
 77 		
 78 		return that["public"];
 79 	};	
 80 	
 81 /**
 82  *  Default event delegation
 83  */	
 84     
 85     // Children
 86 	that.$element.children().each(function(i, e){
 87 		
 88 		var $child = $(e).children();
 89 		
 90 		// Link
 91 		if($child.eq(0).prop("tagName") == "A") {
 92 			$(e).addClass("ch-bellows").children().addClass("ch-bellows-trigger");
 93 			that.children.push( $child[0] );
 94 			return;
 95 		};
 96 		
 97 		// Bellows
 98 		var list = {};
 99 			list.uid = that.uid + "#" + i;
100 			list.type = "bellows";
101 			list.element = e;
102 			list.$element = $(e);
103 			
104 		// Selected -> It can be for example "2" or "2#1"
105 		if(conf.hasOwnProperty("selected")) {
106 			list.open = (typeof conf.selected == "number") ? conf.selected == i : (conf.selected.split("#")[0] == i) ? conf.selected.split("#")[1] : false;
107 		} else {
108 			list.open = false;
109 		};
110 			
111 		that.children.push( ch.bellows.call(list, that) );
112 	});
113     
114     
115 	return that;
116 	
117 };

121 /**
122  * Accordion's content container.
123  * @abstract
124  * @name Bellows
125  * @class Bellows
126  * @augments ch.Navs
127  * @memberOf ch
128  * @param {Configuration Object} conf Object with configuration properties
129  * @return {Chico-UI Object}
130  */
131 ch.bellows = function(controller){
132 
133     /**
134      * Reference to a internal component instance, saves all the information and configuration properties.
135      * @private
136      * @name that
137      * @type {Object}
138      * @memberOf ch.Bellows
139      */ 
140 	var that = this;
141 	
142 	conf = {};
143 	
144 	that.conf = conf;
145 /**
146  *	Inheritance
147  */
148 
149     that = ch.navs.call(that);
150     that.parent = ch.clon(that);
151 	that.controller = controller;
152 	
153 /**
154  *  Protected Members
155  */ 
156 	
157 	that.$container = that.$element.addClass("ch-bellows");
158 	
159 	that.$trigger = that.$container.children(":first");
160 	
161     /**
162      * Component's content.
163      * @public
164      * @name $content
165      * @type {jQuery Object}
166      * @memberOf ch.Bellows
167      */
168 	that.$content = that.$trigger.next();
169     
170     /**
171      * Shows component's content.
172      * @public
173      * @name show
174      * @return {Chico-UI Object}
175      * @memberOf ch.Bellows
176      */
177 	that.show = function(event){
178 
179 		that.prevent(event);
180 
181 		// Toggle
182 		if (that.active) return that.hide(event);
183 
184 		// Accordion behavior (hide last active)
185 		if (!controller.conf.hasOwnProperty("menu")) {
186 
187 			$.each(controller.children, function(i, e){
188 				if (e.tagName == "A") return;
189 				if ( e.hasOwnProperty("active") && e.hasOwnProperty("element") ) {
190 					if (e.active == true && e.element !== that.element) { e.hide(); };
191 				}; 
192 			});
193 			
194 		};
195 		
196 		that.$content.slideDown("fast");
197         
198         that.parent.show(event);
199         
200         return that;
201     };
202     
203     /**
204      * Hides component's content.
205      * @public
206      * @name hide
207      * @return {Chico-UI Object}
208      * @memberOf ch.Bellows
209      */
210     that.hide = function(event){
211     	that.prevent(event);
212     	
213     	// Toggle
214     	if (!that.active) return;
215     	
216     	that.active = false;
217     	
218    		that.parent.hide(event);
219     	that.$content.slideUp("fast");
220 
221         that.$trigger.removeClass("ch-" + that.type + "-on");
222         
223 		if(!ch.utils.html.hasClass("ie6")) that.$content.slideUp("fast");
224 		
225 		that.callbacks("onHide");
226 		
227         return that;
228 	};
229 	
230     /**
231      * Select component's content.
232      * @public
233      * @name select
234      * @return {Chico-UI Object}
235      * @memberOf ch.Bellows
236      */
237 	that.select = function(child) {
238 		that.show();
239 		
240 		// L2 selection
241 		that.$content.find("a").eq( child ).addClass("ch-bellows-on");
242 	};
243 	
244 	
245 /**
246  *  Default event delegation
247  */	 	
248 	
249 	// Closed by default
250 	if(that.open) that.select( parseInt(that.open) ); else that.$content.addClass("ch-hide");
251 	
252 	// Trigger
253 	that.$trigger
254 		.addClass("ch-bellows-trigger")
255 		.bind("click", function(event){ that.show(event) })
256 		.append("<span class=\"ch-arrow\"> »</span>");
257 
258 	// Content
259 	that.$content
260 		.addClass("ch-bellows-content")
261 		.bind("click", function(event){ event.stopPropagation(); });
262 	
263 	
264 	return that;
265 };
