  2 /**
  3  * Cache control utility.
  4  * @abstract
  5  * @name Cache
  6  * @class Cache
  7  * @memberOf ch
  8  */
  9  
 10 ch.cache = {
 11 
 12     /**
 13      * Map of resources cached
 14      * @name map 
 15      * @type {Object}
 16      * @memberOf ch.Cache
 17      */
 18     map: {},
 19     
 20     /**
 21      * Add a resource to the cache control
 22      * @function 
 23      * @name add
 24      * @param {String} url Resource location
 25      * @param {String} data Resource information
 26      * @memberOf ch.Cache
 27      */
 28     add: function(url, data) {
 29         ch.Cache.map[url] = data;
 30     },
 31     
 32     /**
 33      * Get a resource from the cache
 34      * @function
 35      * @name get
 36      * @param {String} url Resource location
 37      * @returns {String} data Resource information
 38      * @memberOf ch.Cache
 39      */
 40     get: function(url) {
 41         return ch.Cache.map[url];
 42     },
 43     
 44     /**
 45      * Remove a resource from the cache
 46      * @function
 47      * @name rem
 48      * @param {String} url Resource location
 49      * @memberOf ch.Cache
 50      */
 51     rem: function(url) {
 52         ch.Cache.map[url] = null;
 53         delete ch.Cache.map[url];
 54     },
 55     
 56     /**
 57      * Clears the cache map
 58      * @function
 59      * @name flush
 60      * @memberOf ch.Cache
 61      */
 62     flush: function() {
 63         delete ch.Cache.map;
 64         ch.Cache.map = {};
 65     }
 66 };
