  2 /**
  3  * TabNavigator UI-Component for static and dinamic content.
  4  * @name TabNavigator
  5  * @class TabNavigator
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11 
 12 ch.tabNavigator = function(conf){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.TabNavigator
 20      */
 21 	var that = this;
 22 
 23 	conf = ch.clon(conf);
 24 
 25 	that.conf = conf;
 26 	
 27 /**
 28  *	Inheritance
 29  */
 30 
 31     that = ch.controllers.call(that);
 32     that.parent = ch.clon(that);
 33 
 34 /**
 35  *  Private Members
 36  */
 37 
 38     /**
 39      * The actual location hash, is used to know if there's a specific tab selected.
 40      * @private
 41      * @name hash
 42      * @type {String}
 43      * @memberOf ch.TabNavigator
 44      */
 45 	var hash = window.location.hash.replace("#!", "");
 46     /**
 47      * A boolean property to know if the some tag should be selected.
 48      * @private
 49      * @name hashed
 50      * @type {Boolean}
 51      * @default false
 52      * @memberOf ch.TabNavigator
 53      */
 54     var hashed = false;
 55     /**
 56      * Get wich tab is selected.
 57      * @private
 58      * @name selected
 59      * @type {Number}
 60      * @memberOf ch.TabNavigator
 61      */
 62     var selected = conf.selected - 1 || conf.value - 1 || 0;
 63     /**
 64      * Create controller's children.
 65      * @private
 66      * @name createTabs
 67      * @function
 68      * @memberOf ch.TabNavigator
 69      */
 70 	var createTabs = function(){
 71 
 72 		// Children
 73 		that.$element.children().eq(0).addClass("ch-tabNavigator-triggers").find("a").each(function(i, e){
 74 
 75 			// Tab context
 76 			var tab = {};
 77 				tab.uid = that.uid + "#" + i;
 78 				tab.type = "tab";
 79 				tab.element = e;
 80 				tab.$element = $(e);
 81 				tab.controller = that["public"];
 82 
 83 			// Tab configuration
 84 			var conf = {};
 85 				conf.open = (selected == i);
 86 				conf.onShow = function(){
 87 					selected = i;
 88 				};
 89 
 90 			// Callbacks
 91 			if ( ch.utils.hasOwn(that.conf, "onContentLoad") ) conf.onContentLoad = that.conf.onContentLoad;
 92 			if ( ch.utils.hasOwn(that.conf, "onContentError") ) conf.onContentError = that.conf.onContentError;
 93 
 94 			// Create Tabs
 95 			that.children.push(
 96 				ch.tab.call(tab, conf)
 97 			);
 98 
 99 			// Bind new click to have control
100 			$(e).unbind("click").bind("click", function(event){
101 				that.prevent(event);
102 				select(i + 1);
103 			});
104 
105 		});
106 
107 		return;
108 
109 	};
110     /**
111      * Select a child to show its content.
112      * @private
113      * @function
114      * @memberOf ch.TabNavigator
115      */
116 	var select = function(tab){
117 
118 		tab = that.children[tab - 1];
119 		
120 		if(tab === that.children[selected]) return; // Don't click me if I'm open
121 
122 		// Hide my open bro
123 		$.each(that.children, function(i, e){
124 			if(tab !== e) e.hide();
125 		});
126 
127 		tab.show();
128 
129         //Change location hash
130 		window.location.hash = "#!" + tab.$content.attr("id");	
131 		
132 		// Callback
133 		that.callbacks("onSelect");
134 		
135         return that;
136 	};
137 
138 /**
139  *  Protected Members
140  */ 
141     /**
142      * The component's content.
143      * @private
144      * @name $content
145      * @type {jQuery Object}
146      * @memberOf ch.TabNavigator
147      */
148 	that.$content = that.$element.children().eq(1).addClass("ch-tabNavigator-content box");
149 
150     
151 /**
152  *  Public Members
153  */
154 
155     /**
156      * The component's instance unique identifier.
157      * @public
158      * @name uid
159      * @type {Number}
160      * @memberOf ch.TabNavigator
161      */
162 	that["public"].uid = that.uid;
163     /**
164      * The element reference.
165      * @public
166      * @name element
167      * @type {HTMLElement}
168      * @memberOf ch.TabNavigator
169      */
170 	that["public"].element = that.element;
171     /**
172      * The component's type.
173      * @public
174      * @name type
175      * @type {String}
176      * @memberOf ch.TabNavigator
177      */
178 	that["public"].type = that.type;
179     /**
180      * Children instances associated to this controller.
181      * @public
182      * @name children
183      * @type {Collection}
184      * @memberOf ch.TabNavigator
185      */
186 	that["public"].children = that.children;
187     /**
188      * Select a specific child.
189      * @public
190      * @function
191      * @name select
192      * @param {Number} tab Tab's index.
193      * @memberOf ch.TabNavigator
194      */
195 	that["public"].select = function(tab){
196 		select(tab);
197 		
198 		return that["public"];
199 	};
200     /**
201      * Returns the selected child's index.
202      * @public
203      * @function
204      * @name getSelected
205      * @returns {Number} selected Tab's index.
206      * @memberOf ch.TabNavigator
207      */	
208 	that["public"].getSelected = function(){ return (selected + 1); };
209 
210 /**
211  *  Default event delegation
212  */	
213 
214     	that.$element.addClass('ch-tabNavigator');
215 
216 	createTabs();
217 
218 	//Default: Load hash tab or Open first tab	
219 	for(var i = that.children.length; i--; ){
220 		if ( that.children[i].$content.attr("id") === hash ) {
221 			select(i + 1);
222 			
223 			hashed = true;
224 			
225 			break;
226 		};
227 	};
228 
229 	return that;
230 	
231 };

235 /**
236  * Simple unit of content for TabNavigators.
237  * @abstract
238  * @name Tab
239  * @class Tab
240  * @augments ch.Navs
241  * @memberOf ch
242  * @param {Configuration Object} conf Object with configuration properties
243  * @returns {Chico-UI Object}
244  */
245 
246 ch.tab = function(conf){
247     /**
248      * Reference to a internal component instance, saves all the information and configuration properties.
249      * @private
250      * @name that
251      * @type {Object}
252      * @memberOf ch.Tab
253      */
254     var that = this;
255 
256 	conf = ch.clon(conf);
257 	conf.icon = false;
258 	
259 	that.conf = conf;
260 
261 	
262 /**
263  *	Inheritance
264  */
265 
266 	that = ch.navs.call(that);
267 	that.parent = ch.clon(that);
268 
269 /**
270  *  Private Members
271  */
272     /**
273      * Creates the basic structure for the tab's content.
274      * @private
275      * @name createContent
276      * @function
277      * @memberOf ch.TabNavigator
278      */
279 	var createContent = function(){
280 		var href = that.element.href.split("#");
281 		var controller = that.$element.parents(".ch-tabNavigator");
282 		var content = controller.find("#" + href[1]);
283 		
284 		// If there are a tabContent...
285 		if ( content.length > 0 ) {
286 			
287 			return content;
288 		
289 		// If tabContent doesn't exists        
290 		} else {
291 			// Set ajax configuration
292 			conf.ajax = true;
293 
294 			// Create tabContent
295 			return $("<div>")
296 				.attr("id", (href.length == 2) ? href[1] : "ch-tab" + that.uid.replace("#","-") )
297 				.addClass("ch-hide")
298 				.appendTo( controller.children().eq(1) );
299 		};
300 
301 	};
302 
303 /**
304  *  Protected Members
305  */ 
306     /**
307      * Reference to the trigger element.
308      * @private
309      * @name $trigger
310      * @type {jQuery Object}
311      * @memberOf ch.Tab
312      */
313 	that.$trigger = that.$element;
314 
315     /**
316      * The component's content.
317      * @private
318      * @name $content
319      * @type {jQuery Object}
320      * @memberOf ch.Tab
321      */	
322 	that.$content = createContent();
323 
324     /**
325      * Process the show event.
326      * @private
327      * @function
328      * @name show
329      * @returns {jQuery Object}
330      * @memberOf ch.Tab
331      */ 
332 	that.show = function(event){
333 		that.prevent(event);
334 
335 		// Load my content if I'need an ajax request 
336 		if( that.$content.html() == "" ) that.$content.html( that.loadContent() );
337 
338 		// Show me
339 		that.parent.show(event);
340 		
341 		return that;
342 	};
343 
344 /**
345  *  Public Members
346  */
347 	
348 	
349 /**
350  *  Default event delegation
351  */
352 
353 	that.configBehavior();
354 
355 	
356 	return that;
357 }
