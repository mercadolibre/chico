  2 /**
  3  * Simple Tooltip UI-Object.
  4  * @name Tooltip
  5  * @class Tooltip
  6  * @augments ch.Floats
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11 
 12 ch.tooltip = function(conf) {
 13     
 14     
 15     /**
 16      * Reference to a internal component instance, saves all the information and configuration properties.
 17      * @private
 18      * @name that
 19      * @type {Object}
 20      * @memberOf ch.Tooltip
 21      */
 22 	var that = this;
 23 	
 24 	conf = ch.clon(conf);
 25 	conf.cone = true;
 26 	conf.content = that.element.title;	
 27 	conf.position = {};
 28 	conf.position.context = $(that.element);
 29 	conf.position.offset = conf.offset || "0 10";
 30 	conf.position.points = conf.points || "lt lb";
 31 	
 32 	that.conf = conf;
 33 	
 34 /**
 35  *	Inheritance
 36  *		
 37  */
 38 
 39     that = ch.floats.call(that);
 40     that.parent = ch.clon(that);
 41 
 42 /**
 43  *  Private Members
 44  */
 45 
 46     
 47 /**
 48  *  Protected Members
 49  */     
 50     that.$trigger = that.$element;
 51 
 52     that.show = function(event) {
 53         that.$trigger.attr('title', ''); // IE8 remembers the attribute even when is removed, so ... empty the attribute to fix the bug.
 54 		that.parent.show(event);
 55 		
 56 		return that;
 57 	};
 58 	
 59     that.hide = function(event) {
 60 		that.$trigger.attr('title', conf.content);
 61 		that.parent.hide(event);
 62 		
 63 		return that;
 64     };
 65 
 66 /**
 67  *  Public Members
 68  */
 69     /**
 70      * The component's instance unique identifier.
 71      * @public
 72      * @name uid
 73      * @type {Number}
 74      * @memberOf ch.Tooltip
 75      */
 76     that["public"].uid = that.uid;
 77     /**
 78      * The element reference.
 79      * @public
 80      * @name element
 81      * @type {HTMLElement}
 82      * @memberOf ch.Tooltip
 83      */
 84     that["public"].element = that.element;
 85     /**
 86      * The component's type.
 87      * @public
 88      * @name type
 89      * @type {String}
 90      * @memberOf ch.Tooltip
 91      */
 92     that["public"].type = that.type;
 93     /**
 94      * The component's content.
 95      * @public
 96      * @name content
 97      * @type {String}
 98      * @memberOf ch.Tooltip
 99      */
100     that["public"].content = that.content;
101     /**
102      * Shows component's content.
103      * @public
104      * @name show
105      * @function
106      * @returns {Chico-UI Object}
107      * @memberOf ch.Tooltip
108      */
109 	that["public"].show = function(){
110 		that.show();
111 
112 		return that["public"];
113 	};
114     /**
115      * Hides component's content.
116      * @public
117      * @name hide
118      * @function
119      * @returns {Chico-UI Object}
120      * @memberOf ch.Tooltip
121      */ 
122 	that["public"].hide = function(){
123 		that.hide();
124 
125 		return that["public"];
126 	};
127     /**
128      * Positioning configuration.
129      * @public
130      * @name position
131      * @memberOf ch.Tooltip
132      */
133 	that["public"].position = that.position;
134 
135 /**
136  *  Default event delegation
137  */	
138  	
139 	that.$trigger
140 		.bind('mouseenter', that.show)
141 		.bind('mouseleave', that.hide);
142 
143     // Fix: change layout problem
144     $("body").bind(ch.events.LAYOUT.CHANGE, function(){ that.position("refresh") });
145 
146 
147 	return that;
148 };
