  2 /**
  3  * Validate numbers.
  4  * @name Number
  5  * @class Number
  6  * @augments ch.Watcher
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties.
  9  * @returns {Chico-UI Object}
 10  * @see ch.Watcher
 11  * @example
 12  * // Create a number validation
 13  * $("input").number("This field must be a number.");
 14  */
 15 
 16 ch.number = function(conf) {
 17 
 18 /**
 19  *  Constructor
 20  */
 21 
 22 	conf = conf || {};
 23 	
 24     conf.messages = conf.messages || {};
 25 
 26     if ( ch.utils.hasOwn(conf, "msg") ) { 
 27 		conf.messages.number = conf.msg;
 28     	conf.msg = null;
 29     	delete conf.msg;
 30     };
 31     
 32     // $.number("message"); support
 33 	if ( !conf.number && !conf.min && !conf.max && !conf.price ) {
 34 		conf.number = true;
 35 	};
 36   
 37 	// Add validation types
 38 	conf.types = "number,min,max,price";
 39     
 40     // Define the conditions of this interface
 41     conf.conditions = [{
 42             name: "number",
 43             patt: /^([0-9\s]+)$/ 
 44     	},{
 45             name: "min",
 46             expr: function(a,b) { return a >= b } 
 47         },{
 48             name: "max",
 49             expr: function(a,b) { return a <= b } 
 50         },{
 51             name: "price",
 52             patt: /^(\d+)[.,]?(\d?\d?)$/ 
 53         }];
 54 
 55 
 56 	return ch.watcher.call(this, conf);
 57     
 58 };

 60 /**
 61  * Validate a number with a minimun value.
 62  * @name Min
 63  * @interface Min
 64  * @augments ch.Number
 65  * @memberOf ch.Number
 66  * @param {Number} value Minimun number value.
 67  * @param {String} [message] Validation message.
 68  * @returns {Chico-UI Object}
 69  * @see ch.Watcher
 70  * @example
 71  * $("input").min(10, "Write a number bigger than 10");
 72  */
 73 
 74 ch.interface( "min", "number" );
