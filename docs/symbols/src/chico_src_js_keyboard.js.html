  2 /**
  3  * Keyboard event controller utility to know wich keys are begin
  4  * @abstract
  5  * @name Keyboard  
  6  * @class Keyboard
  7  * @memberOF ch
  8  * @param {Event Object} event
  9  */ 
 10 ch.keyboard = function(event) {
 11     
 12     /**
 13      * Map with references to key codes.
 14      * @private
 15      * @name keyCodes
 16      * @type {Object}
 17      * @memberOf ch.Keyboard
 18      */ 
 19     var keyCodes = {
 20         "13": "ENTER",
 21         "27": "ESC",
 22         "37": "LEFT_ARROW",
 23         "38": "UP_ARROW",
 24         "39": "RIGHT_ARROW",
 25         "40": "DOWN_ARROW"
 26     };
 27     
 28     if( !ch.utils.hasOwn(keyCodes, event.keyCode) ) return;
 29     
 30     ch.utils.document.trigger(ch.events.KEY[ keyCodes[event.keyCode] ], event);
 31     
 32 };
