  3  *	Chat Component
  4  *  $("#chat").chat({
  5  *      ruleGroupName: "",
  6  *      style: ["block"],
  7  *      template: [1],
  8  *      environment: "1"|"2"|"3"
  9  *  });
 10  */
 11 
 12 ch.chat = function(conf) {
 13     
 14    	var that = ch.object(); // Inheritance
 15 
 16     var getDomain = function(n) {
 17         switch (n) {
 18             case "1": return "mercadolidesa.com.ar"; break;
 19             case "2": return "mercadolistage.com.ar"; break;
 20             case "3": return "mercadolibre.com.ar"; break;
 21         }
 22     }
 23 
 24     if (conf.msg) {
 25         conf.ruleGroupName = conf.msg;
 26     }
 27 
 28     that.load = function() {
 29         loadChatGZ(conf.ruleGroupName, conf.element.id, conf.style||"block", conf.template||"1",conf.environment||"3"); 
 30     }
 31 
 32    	ch.get({
 33    	    method: "component",
 34    	    name: "chat",
 35    	    script: "http://www."+getDomain(conf.environment)+"/org-img/jsapi/chat/chatRBIScript.js",
 36    	    callback: function() {
 37        	    that.load(); 
 38         }
 39    	});
 40 
 41     that["public"] = {
 42     	uid: conf.uid,
 43 		element: conf.element,
 44         type: "chat"
 45     }
 46     
 47     return that;
 48 
 49 }
